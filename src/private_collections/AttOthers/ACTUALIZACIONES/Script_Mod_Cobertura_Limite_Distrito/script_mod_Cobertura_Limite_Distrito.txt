_block
    ruta_fuente <<"C:\Users\smladmin\Documents\cambios_masivos\test.txt"  # Ruta del archivo fuente
    ruta_logs <<"C:\Users\smladmin\Documents\cambios_masivos\logs"  # Ruta donde se guardarán los logs
    ruta_exitosos << ruta_logs + "\exitosos_log.txt"
    ruta_fallidos_general << ruta_logs + "\fallidos_general_log.txt"
    ruta_fallidos_tipo_incorrecto << ruta_logs + "\fallidos_tipo_log.txt"
    ruta_fallidos_distrito << ruta_logs + "\fallidos_distrito_log.txt"
    vista << gis_program_manager.cached_dataset(:gis)  # Acceso al dataset GIS
    #vista.checkpoint("Modificando tipo_cobertura " + date_time.now().write_string)
    limite_dist << vista.collection(:limite_distrito_cobre)
    tipos_validos << rope.new_with("MSAN", "URBA", "VIRTUAL")  # Tipos de cobertura válidos
    registros_exitosos << rope.new()
    fallos_distrito << rope.new()
    fallos_tipo_incorrecto << rope.new()
    registros_fallo_update << rope.new()
    lineas_error << rope.new()
    archivo_fuente << external_text_input_stream.new(ruta_fuente)
    contador << 0
    _for una_linea _over 1.upto(100000000)  # Comienza un bucle para procesar las líneas del archivo fuente
    _loop
        _try
            linea << archivo_fuente.get_line()
            _if linea _is _unset _then  # Si la línea está vacía, se detiene el procesamiento
                show("Total de líneas procesadas", contador)
                archivo_fuente.close()
                _leave
            _endif
            contador +<<1  # Incrementamos el contador de líneas procesadas
            _if contador _mod 50 = 0  # Cada 50 líneas procesadas, muestra el progreso
            _then
                show("Líneas procesadas: ", contador, " a ", date_time.now().write_string)
            _endif
            texto << linea.write_string.split_by(";", _true )  # Separamos la línea por el delimitador ";"
            _if texto.size <> 2  # Aseguramos que la línea tenga 2 partes
            _then
                fallos_distrito.add(linea)  # Si no tiene 2 partes, la línea se considera mal formada
                _continue
            _endif
            id_distrito << texto[1].as_number()  # Extraemos el ID del distrito
            tipo_cobertura << texto[2].write_string  # Extraemos el tipo de cobertura
            un_distrito << limite_dist.select(predicate.eq(:id, id_distrito)).an_element()  # Buscamos el distrito en la tabla
            write(un_distrito)
            _if un_distrito _is _unset  # Si el distrito no existe, lo marcamos como fallo
            _then
                fallos_distrito.add(linea)
                _continue
            _endif
            # Validación del tipo de cobertura
            _if _not (tipos_validos.eq_includes?(tipo_cobertura))  # Si el tipo no es válido
            _then
                fallos_tipo_incorrecto.add(linea)  # Se agrega al registro de fallos por tipo incorrecto
                _continue
            _endif
            # Actualización del campo tipo_cobertura
            write("tipo_cobertura", %tab, tipo_cobertura)
            pl_free << property_list.new_with(:tipo_cobertura, tipo_cobertura)  # Se crea una lista de propiedades con el nuevo valor para tipo_cobertura
            r << record_transaction.new_update(un_distrito, pl_free).run()  # Se realiza la actualización en la tabla
            registros_exitosos.add(linea)  # Si la actualización fue exitosa, la línea se agrega a los registros exitosos
        _when error  # Si ocurre un error durante la actualización
            registros_fallo_update.add(linea)  # Se agrega la línea al registro de fallos
            _continue
        _endtry
    _endloop
    archivo_fuente.close()  # Cerramos el archivo fuente después de procesar todas las líneas
    vista.commit()  # Confirmamos los cambios realizados en el sistema de base de datos
    # Generación de logs de éxito
    _if registros_exitosos.size <> 0
    _then
        archivo_exitosos << external_text_output_stream.new(ruta_exitosos)
        archivo_exitosos.write("Líneas exitosamente actualizadas.")
        archivo_exitosos.newline()
        archivo_exitosos.write("_______________________________________________________________")
        archivo_exitosos.newline()
        archivo_exitosos.write("TOTAL: ", registros_exitosos.size)
        archivo_exitosos.newline()
        archivo_exitosos.newline()
        _for exito _over registros_exitosos.elements()  # Escribimos las líneas exitosas en el archivo de logs
        _loop
            archivo_exitosos.write(exito.write_string)
            archivo_exitosos.newline()
        _endloop
        archivo_exitosos.close()
    _endif
    # Generación de logs de fallos generales
    _if fallos_distrito.size <> 0
    _then
        archivo_fallidos_general << external_text_output_stream.new(ruta_fallidos_general)
        archivo_fallidos_general.write("Líneas con fallos en el distrito.")
        archivo_fallidos_general.newline()
        archivo_fallidos_general.write("_______________________________________________________________")
        archivo_fallidos_general.newline()
        archivo_fallidos_general.write("TOTAL: ", fallos_distrito.size)
        archivo_fallidos_general.newline()
        _for fallo_distrito _over fallos_distrito.elements()  # Escribimos los fallos por distrito en el archivo de logs
        _loop
            archivo_fallidos_general.write(fallo_distrito.write_string)
            archivo_fallidos_general.newline()
        _endloop
        archivo_fallidos_general.close()
    _endif
    # Generación de logs de fallos por tipo de cobertura incorrecto
    _if fallos_tipo_incorrecto.size <> 0
    _then
        archivo_fallidos_tipo_incorrecto << external_text_output_stream.new(ruta_fallidos_tipo_incorrecto)
        archivo_fallidos_tipo_incorrecto.write("Tipo de cobertura no válido.")
        archivo_fallidos_tipo_incorrecto.newline()
        archivo_fallidos_tipo_incorrecto.write("_______________________________________________________________")
        archivo_fallidos_tipo_incorrecto.newline()
        archivo_fallidos_tipo_incorrecto.write("TOTAL: ", fallos_tipo_incorrecto.size)
        archivo_fallidos_tipo_incorrecto.newline()
        _for fallo_tipo _over fallos_tipo_incorrecto.elements()  # Escribimos los fallos por tipo incorrecto en el archivo de logs
        _loop
            archivo_fallidos_tipo_incorrecto.write(fallo_tipo.write_string)
            archivo_fallidos_tipo_incorrecto.newline()
        _endloop
        archivo_fallidos_tipo_incorrecto.close()
    _endif
_endblock
