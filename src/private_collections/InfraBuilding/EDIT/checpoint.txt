_pragma(classify_level=basic, topic={swg,swg_design,dsn_admin})
## This is a specific permanently residing non-instantiable database
## manager class for the SWS Design Manager product. It controls the
## access to the database alternatives and versions created and
## used by the planning system within the application datasets.
## After the planning system having been installed the standard
## version_form_editor should no longer be active. While planning 
## the design administration is responsible for the maintenance of
## planning alternatives (as a client of swg_dsn_user_alternative_control)
## whereas in normal GIS mode a swg_dsn_version_form_editor should
## work instead of the standard version_form_editor in order to hide
## the planning alternatives.
## The swg_dsn_user_alternative_control has protocols for
## following services:
## 1. In planning mode:
## - Alternatives corresponding to a given scheme hierarchy position must be
##   simultaneously created, retrieved, maintained and controlled in a set of
##   datasets;
## - for alternatives associated to a scheme in the hierarchy corresponding
##   reference versions related to the situation before planning are to
##   maintained;
## - record changes with respect to the relevant reference versions must be
##   detected and yielded;
## - Changes in ancestor alternatives with respect to the active ones are to
##   be watched;
## - Replicates of the reference versions must be provided when requested and
##   administrated when released.
## 2. In normal GIS mode:
## - For any given alternative of a dataset it must be determined whether it
##   related to the planning system or not;
## 3. Always:
## - Indicates whether the planning system is working or not (therefore the
##   front-end of the product SWS Design Manager must announce its start and
##   quitting here.

_block
	## The following two globals,!RELOCATING_SCHEME_ID! and
	## !RELOCATING_TEMP_SCHEME_ID!, are used for resolving the
	## conflict of mapping two schemes, an original scheme (having
	## scheme ID !RELOCATING_SCHEME_ID!) whose alternatives are
	## being relocated to a new, non-pool managed, location in the
	## design alternative hierarchy and a temporary scheme (having
	## scheme ID !RELOCATING_TEMP_SCHEME_ID!) created and
	## associated with the new alternatives whilst a far-merge
	## process between the original scheme and the temporary scheme
	## is used to transfer the scheme changes to their new
	## location. On completion of the transfer, the temporary
	## scheme is deleted and the original scheme refers to the new
	## alternatives by the default behaviour that the original
	## scheme id is encoded into the alternative name. These
	## dynamics are therefore set, or not, to indicate to methods
	## ID_TO_ALT_NAME() and ALT_NAME_TO_ID() whether the
	## alternative are to map to temporary scheme (if dynamics
	## are set for both these variable) or (if the dynamics are not
	## set), to the relocated scheme on completion of the relocation.

	_global !relocating_scheme_id!
	_global !relocating_temp_scheme_id!
_endblock

	## <dataset_name> : symbol (the internal name of a

	##                         for the keyed dataset
	##                         <dataset_name>);
	## <free??> : kleenean (indicates, whether <replicate>

	##                         or discarded:
	##                         _false = occupied,
	##                         _true = free,

	##                            request which suggests the prior
	##                            rendering having been aborted.)

#------------------------------------------------------------------------------
# method design_system_working?<<   KGU
#------------------------------------------------------------------------------
_pragma(classify_level=advanced, topic={swg,swg_design,dsn_user_alt_ctrl})
_method swg_dsn_user_alternative_control.design_system_working?<< working?
	## D e s c r i p t i o n :
	## This method is the internal interface for the Design Assistant to
	## inform the User Alternative Control about its activation/deactivation.
	## Subsequent calls of method design_system_working? will return the
	## value WORKING.
	## When WORKING? is _true, then also an initialisation is done. When it
	## is _false, then cached replicates and other information no
	## longer needed is dropped. 
	##
	## A r g u m e n t s  (- <name>: <class>; <description>):
	## - WORKING?: boolean; indicates, whether the Design Manager system
	##   is being activated (_true) oder deactivated (_false).
	##
	## D y n a m i c   V a r i a b l e s  (- <name>: <class>; <description>):
	## none
	##
	## R e s u l t s  (- <name>: <class>; <description>):
	## - WORKING?: boolean; same as given.
	##
	## S i d e   E f f e c t s  (incl. important slot changes):
	## .design_system_working? is assigned the value of WORKING?;
	## .replicates are cleared.
	## If WORKING? is _true and .design_system_working? was _false,
	## then an initialisation is done, i.e. also slots .dataset_infos,
	## .replicates, and .view_status will be set appropriately where
	## the current alternatives of several datasets may change to
	## the configured design top versions.
	## If WORKING? is _false, however, and .design_system_working?
	## was _true, then slots .view_status and .replicates will be
	## emptied (after remaining replicates having been discarded).
	##
	## C o n d i t i o n s  (- <name>: <type>; <indication>):
	## - swg_dsn_design_view_not_found: swg_dsn_error; if the design
	##   administration view is not found (with WORKING? = _true only);
	## - ds_view_updated: user_error; if one of the datasets
	##   relevant for planning contains unsaved changes. 
	## - db_thing_not_found: swg_dsn_error; if the as-built data alternative
	##   is not found in the given DATASET or the dataset was not found.
	## - swg_dsn_standard_version_menu_found: swg_dsn_information; if an
	##   active instance of class version_form_editor was detected.

	_local dataset # database_view
	_local ds_info # swg_dsn_dataset_info (record)
			
	_if working? _isnt .design_system_working?
	_then
		_if working?
		_then
			_self.initialise()
		_else
			#
			# It is a switch-off: Reset the design system data
			# and leave the planning alternatives
			#
			# Discard all eventually cached replicates and empty the
			# replicate cache.
			_self.remove_reference_replicates()

			# Leave the planning environment
			_for ds_entry _over .view_status.fast_elements()
			_loop
				# First, get the referred dataset
				dataset << gis_program_manager.
					   cached_dataset(ds_entry.dataset_name)
				_if dataset _is _unset _then _continue _endif 
			# ^-------------------------------------------

				# Now the as-built data alternative path is needed
				ds_info << .dataset_infos.select(
						   predicate.eq(
							   :dataset_name,
							   ds_entry.dataset_name.as_charvec()
							     )
							  ).an_element()
				_if ds_info _is _unset _then _continue _endif 
			# ^-------------------------------------------

				# Change the alternative, drop eventually unsaved data before
				# (this should not happen, but a condition is to be avoided).
				_if dataset.updated?
				_then
					dataset.rollback()
				_endif
				dataset.go_to_alternative(ds_info.as_built_alternative_path,
							  :readonly)
				dataset.remove_dependent(_self)
			_endloop
			.effective_scheme_path << _unset
		_endif
	_endif

	>> (.design_system_working? << working?)
	
_endmethod
$

#------------------------------------------------------------------------------
# Method: is_design_alternative?()   KGU
#------------------------------------------------------------------------------
_pragma(classify_level=basic,
	topic={swg,swg_design,dsn_user_alt_ctrl, database_version_management})
_method swg_dsn_user_alternative_control.is_design_alternative?(dataset_name,alternative_path)
	## D e s c r i p t i o n :
	## Returns a Boolean result being _true if and only if the given
	## alternative path refers to an alternative controlled by the
	## Design Manager system for schemes. This includes the "design top",
	## the "build-plan", and all descendent alternatives. The decisive
	## information is taken from the swg_dsn_dataset_info collection of
	## the Design Administration datasets.
	## 
	## A r g u m e n t s  (- <name>: <class>; <description>):
	## - DATASET_NAME: symbol; the (internal) name of a dataset;
	## - ALTERNATIVE_PATH: list of string; must be the list (rope) of
	##     alternative names down from top (in the format yielded by
	##     method database_view.alternative_path).
	## 
	## D y n a m i c   V a r i a b l e s  (- <name>: <class>; <description>):
	## none
	## 
	## R e s u l t s  (- <name>: <class>; <description>):
	## - IS_DESIGN_ALTERNATIVE?: boolean; will be _true if the given
	##     alternative path in the given dataset refers to an alternative
	##     controlled by SWS Design Manager.
	## 
	## S i d e   E f f e c t s   (incl. important slot changes):
	## none
	## 
	## C o n d i t i o n s  (- <name>: <type>; <indication>):
	## - swg_dsn_uac_badly_initialised: swg_dsn_warning; if the User
	##     Alternative Control is not ready to identify an
	##     alternative path as design alternative,
	## - no_coercion_defined: error; if the given arguments are of
	##     inappropriate type.

	_local is_design_alternative? << _false  # boolean; the default result
	_local as_built_path   # rope of string; the path of the as-built data alternative,
	                       # but with preceding _unset element.
	_local ds_info         # swg_dsn_dataset_info; the respective information record
	_local below_as_built  # byte_character_vector; an alternative name

	_local catch_dataset << "no_admin_view"

	# Check the argument type first
	:symbol.check_type(dataset_name)
# <= = = = = = = = = = = = = = = = = = =
	_if _not alternative_path.responds_to?(:|[]|)
	_then
		# Somewhat senseless, but required by the style guide.
		condition.raise(:does_not_understand, :object,alternative_path, :selector, :|[]|,
				:arguments,{},:iterator?,_false,:private?, _false)
# <===============================================================================
	_endif

	_catch catch_dataset
	       _handling swg_dsn_design_view_not_found, swg_dsn_no_dataset_info_found _with
	       _proc @no_admin(cond)
		       _import catch_dataset
		       condition.raise(:information,:string,cond.report_contents_string)
		       _throw catch_dataset
	       _endproc
		
	       _if .dataset_infos _is _unset
	       _then
		       # _self has not been initialised yet. Try to do the most urgent
		       # now: First find the design administration dataset (this may raise
		       # a condition swg_dsn_design_view_not_found!) and then set the slot
		       _self.set_dataset_infos()
	# <= = = = = = = = = = = = = = = = = = =
		       _if .dataset_infos _is _unset 
		       _then
			       # If the slot initialisation failed, give a warning that the
			       # UAC has dubious initialisation status and skip.
			       condition.raise(:swg_dsn_uac_badly_initialised)
			       _throw catch_dataset
	# <========================================
		       _endif
	       _endif

	       ds_info << .dataset_infos.select(predicate.eq(:dataset_name,dataset_name)).an_element()
	       _if ds_info _isnt _unset _andif
		   (as_built_path << _self.decompose_alternative_name(
					     ds_info.as_built_alternative_path)) _isnt _unset _andif
		   # Only if the as-built path (note that it is preceded with an _unset)
		   # is shorter than the given alternative_path, then it can be a prefix...
		   as_built_path.size.in_range?(1,alternative_path.size) _andif
		   as_built_path.first _is _unset
	       _then
		       # Now check whether the as_built_path really is a prefix of the
		       # given path (remember we must skip the _unset).
		       _for level _over 2.upto(as_built_path.size)
		       _loop
			       _if as_built_path[level].lowercase ~= alternative_path[level-1].lowercase
			       _then
				       _leave
			# <==================
			       _endif
		       _finally
			       # The as-built is a prefix, it remains to check for design top
			       # or build-plan top alternative
			       below_as_built << alternative_path[as_built_path.size]
			       is_design_alternative? <<
				       below_as_built = ds_info.design_top_alternative_name _orif
				       below_as_built = ds_info.build_plan_alternative_name
		       _endloop
	       _endif
	       
	_endcatch #catch_dataset

	>> is_design_alternative?
	
_endmethod
$

#------------------------------------------------------------------------------
# Method: create_scheme_environment()   KGU
#------------------------------------------------------------------------------
_pragma(classify_level=restricted,
	topic={swg, swg_design, dsn_admin, dsn_user_alt_ctrl})
_pragma(classify_level=restricted, topic={swg, swg_design, dsn_admin, dsn_user_alt_ctrl})
_method swg_dsn_user_alternative_control.create_scheme_environment(scheme_path, dataset_infos, _gather keys_and_args)

	## D e s c r i p t i o n :
	## Creates an appropriate alternative per dataset specified
	## by a dataset information record contained in DATASET_INFOS
	## for the given key sequence SCHEME_PATH.
	## First checks that all specified datasets are accessible (if
	## not, then a warning is raised, see below).
	## If necessary, missing ancestors in the alternative path will
	## be completed.
	## For all alternatives created, a corresponding start version
	## is fixed in an appropriate way for further reference.
	##
	## If SCHEME_PATH is a path for a top-level scheme, the
	## alteratives for it will be requested from the alternative
	## pool manager.
	##
	## In all specified datasets where the required alternative was
	## creatable this is gone to, in the remaining ones (if
	## specified) to the deepest existing alternative on the path
	## (in readonly mode).
	## The environment of datasets not specified remains unchanged.
	##
	## The maximum alternative level defined in constant max_alt_level
	## must not be exceeded.
	## 
	## A r g u m e n t s  (- <name>: <class>; <description>):
	## - SCHEME_PATH: list of sys_id; Key sequence representing the
	##     design tree path from the root (base scheme) down to the
	##     very scheme the alternatives are to be created for;
	##     A rope with first element 0, however, designates a
	##     build-plan pool alternative;
	## - DATASET_INFOS: collection of swg_dsn_dataset_info; a set
	##     of dataset information records specifying the datasets
	##     alternatives are to be created for.
	## - KEYS_AND_ARGS: alternating key/value pairs; Optional list
	##     of creation parameters, to be passed to the dataset
	##     alternative creation method, for each dataset associated
	##     with the scheme.
	##     Refer to method CREATE_MAPPED_ALTERNATIVE() for details
	##     of valid options.

	## - KEYS_AND_ARGS: alternating key/value pairs; Optional list
	##     of additional parameters. Valid options are those which
	##     may be to be passed to the dataset alternative creation
	##     method, for each dataset associated with the scheme
	##     (refer to method CREATE_MAPPED_ALTERNATIVE() for details
	##     of those options), along with the following additional
	##     options used local to this method:
	##        :relocating_scheme_id, integer
	##           If SCHEME_PATH is a path to a temporary scheme
	##           created to facilitate the relocation of the
	##           alternatives of another scheme to another
	##           (non-pool managed) location in the design
	##           hierarchy (using a far-merge operation to the
	##           temporary scheme) then this option must be 
	##           set to the ID of the scheme that is being
	##           relocated.
	## 
	## D y n a m i c   V a r i a b l e s  (- <name>: <class>; <description>):
	## none
	## 
	## R e s u l t s  (- <name>: <class>; <description>):
	## - CREATION_INFO: property_list[symbol] of boolean; For each of
	##     the dataset_names referred to by the given DATASET_INFOS
	##     records, a boolean entry reports whether the requested
	##     alternative is available (_true) or not (_false).
	## 
	## S i d e   E f f e c t s   (incl. important slot changes):
	## Slot .view_status will be updated.
	## 
	## C o n d i t i o n s  (- <name>: <type>; <indication>):
	## - swg_dsn_design_system_not_working: swg_dsn_error; if the design
	##     system has not been started or was switched off;
	## - swg_dsn_missing_dataset: swg_dsn_warning; if required datasets are
	##     not found;
	## - swg_dsn_scheme_modified: swg_dsn_warning; when the required alternative
	##     change would fail because of unsaved changes.
	## - swg_dsn_max_alternative_level: swg_dsn_error; if the creation of an
	##     alternative would exceed the level value _self.max_alt_level.

	# Ignore the warning that a design alternaive is missing and needs to be created as this
	# is to be expected _when we are initially creating the design, rather than re-activating it.
											      
	_handling swg_dsn_missing_alternative_do_create _with _proc() _endproc 
	
	creation_info << _self.int!go_to_scheme_environment(scheme_path, dataset_infos, :write, _scatter keys_and_args)

	>> creation_info
_endmethod
$

#------------------------------------------------------------------------------
# Method: go_to_scheme_environment()
#------------------------------------------------------------------------------
_pragma(classify_level=restricted, topic={swg, swg_design, dsn_admin, dsn_user_alt_ctrl})
_method swg_dsn_user_alternative_control.go_to_scheme_environment(scheme_path, dataset_infos, wanted_mode, _gather keys_and_args)
	## D e s c r i p t i o n :
	## The dataset environment is arranged so that the dataset
	## views referred to by the given DATASET_INFOS records are
	## moved to the alternatives associated with the given SCHEME_PATH.
	## If there are no such alternatives then first a warning is raised
	## and then, according to the value of ADD_MISSING_VERSIONS?, they
	## are either added or the deepest existing alternative on the path
	## is gone to with readonly mode.
	## If the former version environment contains unsaved data,
	## then a warning is raised, and if ignored, then all changes
	## are committed.
	##
	## For each dataset for which the scheme alternative was
	## successfully reached, the dataset will be checked to see if
	## there was a previous operation run on the scheme that
	## resulted in a failure such that the scheme is no longer
	## usueable until a successful recovery operation has been
	## performed on the dataset (note that the dataset check is a
	## lightwight operation) and if detected the recovery operation
	## will be run on the dataset. If any recovery operation failse
	## then all of the datsets specified by dataset_infos will be
	## switched to readonly and an error condition raised.
	##
	## If SCHEME_PATH is empty then the access mode in all
	## requested datasets will at most be :readonly.
	##
	## If SCHEME_PATH is a path for a top-level scheme, the
	## alteratives for it will be requested from the alternative
	## pool manager.
	## 
	## A r g u m e n t s  (- <name>: <class>; <description>):
	## - SCHEME_PATH: list of sys_id; Key sequence representing the
	##     design tree path from the root (base scheme) down to the
	##     very scheme the alternatives for which are to be gone to;
	##     A rope with first element 0, however, designates a
	##     build-plan pool alternative;
	## - DATASET_INFOS: collection of swg_dsn_dataset_info; a set
	##     of dataset information records specifying the datasets
	##     within which appropriate alternatives are to be dteetced
	##     and gone to;
	## - WANTED_MODE: symbol; One of :write or :readonly.
	## - KEYS_AND_ARGS: alternating key/value pairs; Optional list
	##     of additional parameters. Valid options are those which
	##     may be to be passed to the dataset alternative creation
	##     method, for each dataset associated with the scheme
	##     (refer to method CREATE_MAPPED_ALTERNATIVE() for details
	##     of those options), along with the following additional
	##     option used local to this method:
	##        :add_missing_versions?, boolean
	##           Defines whether missing design alternatives are to
	##           be created and added to the design (wherever
	##           possible), after a corresponding warning has been
	##           issued. 
	##           Default is _TRUE, to attempt to add any missing
	##           versions.
	## 
	## D y n a m i c   V a r i a b l e s  (- <name>: <class>; <description>):
	## none
	## 
	## R e s u l t s  (- <name>: <class>; <description>):
	## - ACCESS_MODE: property_list[symbol] of symbol; for every
	##     dataset name (key) known to the Design Manager (all configured
	##     dataset information records) there will be an entry :none
	##     (if no access at all is possible, e.g. dataset not found
	##     or the design alternative within the dataset is missing
	##     and could not be created), :readonly (if there is no
	##     write access, e.g. because the dataset is not associated
	##     or write access was not wanted), or :write (if write
	##     access was requested and is available).
	## 
	## S i d e   E f f e c t s   (incl. important slot changes):
	## The new version environment is cached in .view_status, including
	## the corresponding design-start version replicates.
	## 
	## C o n d i t i o n s  (- <name>: <type>; <indication>):
	## - swg_dsn_design_system_not_working: swg_dsn_error; if the design
	##     system has not been started or was switched off.
	## - swg_dsn_missing_dataset: swg_dsn_warning; if requested
	##     datasets are not found;
	## - swg_dsn_missing_alternative_do_create: swg_dsn_warning; if
	##     in a requested dataset the according alternative(s) are
	##     missing and the value of the shared variable
	##     swg_dsn_admin_engine.create_missing_versions_on_design_open?
	##     is _true.
	## - swg_dsn_missing_alternative_create_failed: swg_dsn_warning; if
	##     in a requested dataset the according alternative(s) are
	##     missing and could not be created.
	## - swg_dsn_missing_alternative_skip_create: swg_dsn_warning; if
	##     in a requested dataset the according alternative(s) are
	##     missing and the value of the shared variable
	##     swg_dsn_admin_engine.create_missing_versions_on_design_open?
	##     is _false.
	## - swg_dsn_scheme_modified: swg_dsn_warning; if, going to leave
	##     the current alternative, unsaved changes are found in
	##     some dataset.
	## - swg_dsn_vbase_merge_failure_detected: swg_dsn_warning; if
	##     a required dataset error recovery operastion was detected. 
	## - swg_dsn_recovering_failed_vbase_merge: swg_dsn_information;
	##     if a required dataset error recovery operastion is being
	##     run. 
	## - swg_dsn_error_recovery_completed: swg_dsn_information; if
	##     if a required dataset error recovery operastion was
	##     successfully completed.
	## - swg_dsn_error_recovery_failed: swg_dsn_error; if a
	##     required error recovery operation was detected for one
	##     of the scheme datasets but when run the recovery failed
	##     to complete.

	(found?, access_mode) << _self.int!go_to_scheme_environment(scheme_path, dataset_infos, wanted_mode, _scatter keys_and_args)

	_if (sch << _self.scheme(scheme_path)) _isnt _unset
	_then 
	# Check each dataset for any error recovery required and run it
	# if needed (NOTE: The checks made here must be extremely lightweight)
		_for ds_info _over dataset_infos.fast_elements()
		_loop
			_if (ds << ds_info.referred_dataset) _isnt _unset _andif found?[ds_name << ds.name] _is _true
			_then 
				_if (recovery_process_list << ds.swg_dsn_detect_error_recovery(sch)) _isnt _unset
				_then
					ok? << _true
					_if error_recovery_ok? _is _unset _then error_recovery_ok? << _true _endif 
					_try _with cond
						ds.swg_dsn_do_error_recovery(sch, recovery_process_list)
					_when error
						_if err_reason _is _unset
						_then
							err_reason << cond.report_contents_string
							ds_name << ds.original_view_name 
						_endif 
						ds.rollback()
						ds.switch(:readonly)
						ok? << _false 
					_endtry 
					error_recovery_ok? << error_recovery_ok? _and ok?
				_endif
			_endif 
		_endloop
		_if error_recovery_ok? _is _false 
		_then
		# Ensure all scheme datasets are readonly and raise an error
		# that the error recovery failed to complete.
			_for ds_info _over dataset_infos.fast_elements()
			_loop
				_if (ds << ds_info.referred_dataset) _isnt _unset _andif ds.mode _isnt :readonly
				_then
					ds.switch(:readonly)
				_endif 
			_endloop 
			condition.raise(:swg_dsn_error_recovery_failed, :project_and_design_name, sch.project_and_scheme_name,
					:dataset, ds_name, :reason, err_reason)
		_elif error_recovery_ok? _is _true
		_then 
		# Inform successful recovery
			condition.raise(:swg_dsn_error_recovery_complete, :project_and_design_name, sch.project_and_scheme_name)
		_else
		# Error ecovery was not required
		_endif 
	_endif 
	>> access_mode
_endmethod
$

#------------------------------------------------------------------------------
# Method: drop_scheme_environment()   KGU
#------------------------------------------------------------------------------
_pragma(classify_level=restricted,
	topic={swg, swg_design, dsn_admin, dsn_user_alt_ctrl})
_pragma(classify_level=restricted, topic={swg, swg_design, dsn_admin, dsn_user_alt_ctrl})
_method swg_dsn_user_alternative_control.drop_scheme_environment(scheme_path, dataset_infos,
								 _optional recursively?, relocating_scheme_id)
	## D e s c r i p t i o n :
	## Removes the alternatives associated to the given SCHEME_PATH
	## (unless the scheme alternatives are pool managed, see below)
	## in all datasets described by the dataset information records
	## contained in DATASET_INFOS. As far as there are checkpoints
	## referring to the deleted alternatives in other alternatives
	## these will also be cancelled.
	##
	## If SCHEME_PATH is a path for a top-level scheme, and its
	## altrnatives are being pool managed then, instead of the
	## alternatives being removed they will be marked as free by
	## the pool manager and retained for re-use by a new design.
	## For top-level designs that are not currently pool managed
	## (i.e. those designs created prior to the release of the pool
	## manager functionality) the alternative will still be
	## retained and added a new free alternative to the pool.
	##
	## If one ore more of the required datasets are missing, a
	## prior warning will be induced.
	##
	## If there are sub-alternatives, then the effect depends on
	## the argument RECURSIVELY?: with _true, all sub-alternatives
	## are also removed, with _false (the default), an error
	## condition will rise.
	##
	## A r g u m e n t s  (- <name>: <class>; <description>):
	## - SCHEME_PATH: list of integer; a sequence of swg_dsn_scheme
	##     keys, according to the scheme path down from root (top
	##     scheme) to the very scheme; 
	##     a rope with first element 0, however, designates a build-
	##     plan pool alternative;
	## - DATASET_INFOS: collection of swg_dsn_dataset_info; the
	##     dataset information records describing those dataset,
	##     within which associated alternatives are to be removed;
	## - RECURSIVELY?: boolean; should be _true in order to remove
	##     eventual sub-alternatives without warning. Otherwise in
	##     this case an error condition would rise. Default: _false;
	## - RELOCATING_SCHEME_ID, integer
	##     If SCHEME_PATH is a path to a temporary scheme created
	##     to facilitate the relocation of the alternatives of
	##     another scheme to another (non-pool managed) location in
	##     the design but is now being dropped because the
	##     relocation failed and is non-recoverable then then this
	##     option must be set to the ID of the scheme that was being
	##     relocated as the alternatives, which are no longer
	##     required, are default mapped to that scheme.
	## 
	## D y n a m i c   V a r i a b l e s  (- <name>: <class>; <description>):
	## none
	## 
	## R e s u l t s  (- <name>: <class>; <description>):
	## - DROP_INFO: property_list[symbol] of kleenean; for every
	##     dataset name (as key) contained in DATASET_INFOS there
	##     will be an entry with following meaning:
	##     _true: removal completely successful (or not necessary);
	##     _false: no removal (e.g. because of missing dataset);
	##     _maybe: partial removal
	## 
	## S i d e   E f f e c t s   (incl. important slot changes):
	## 
	## C o n d i t i o n s  (- <name>: <type>; <indication>):
	## - swg_dsn_design_system_not_working: swg_dsn_error; if the design
	##     system has not been started or was switched off.
	## - swg_dsn_missing_dataset: swg_dsn_warning; if a requested
	##     dataset is not open or not found;
	## - swg_dsn_scheme_occupied: swg_dsn_error; if one of the
	##     alternatives to be removed is occupied by some user (this
	##     might be the current user himself).
	## - ds_alternative_has_sub_alternatives: user_error; if in one of
	##     the datasets sub-alternatives are found, but RECURSIVELY
	##     is not _true.
	## - swg_dsn_information: information; when despite of all prior
	##     checking an access conflict occurred on the run. Parameter
	##     :string contains the dataset and the error message.
	#
	# A first approach to lock all affected alternatives via
	# writable replicates was refrained from because of the
	# fact that replicates are too pcl-consuming. In any case the
	# replicates could not prevent other users from going to these
	# alternatives with read access which is as harmful as having a
	# current writer for the removal of an alternative. So we must
	# try and then look how to get rid of errors, anyway. Thus,
	# a successive removal in post-order is performed (which
	# may skip to the next dataset when faced with an access
	# conflict).
	
	_dynamic !relocating_scheme_id!           # Set to the value of the optional parameter relocating_scheme_id
		                                  # If set, this will be the ID of the scheme being relocated via the
		                                  # temporary scheme that has been specified by SCHEME_PATH
		                                  # with ID !relocating_temp_scheme_id! (see method comment above)
	_dynamic !relocating_temp_scheme_id!      # If !relocating_scheme_id! has been set (see above) then this
		                                  # will be set to the ID of the temporary scheme specified by SCHEME_PATH,
						  # i.e. the value of SCHEME_PATH.LAST
	_local ds_view                            # database_view; the current dataset
	_local cp_name                            # byte_character_vector; a checkpoint name
	_local drop_info <<                       # property_list[symbol] of kleenean; result
		property_list.new(dataset_infos.size)   
	_local at_target_version?                 # boolean; notes whether the requested
	                                          # alternative is reached.
	_local temp_repl                          # ds_version_view; a temporal replicate
	_local alt_path                           # rope of (_unset/string); a mapped alternative path
	_local top_scheme? <<                     # _true if scheme_path is that of top-level scheme
		_self.top_level_scheme?(scheme_path)
	_local non_dc_managed_dsinfos <<          # subset for those which are not Design Clerk managed
	        _self.non_dc_managed_dataset_infos(dataset_infos)
	
	# Check the existence and accessibility of the required datasets
	# - scout: property_list[symbol] of boolean; dataset seek results:
	#     _false means that the dataset is missing;
	# - missing_string: character_vector; names of missing datasets;
	# - unsaved?: boolean; flag for unsaved changes.
	# May cause an error condition swd_dsn_design_system_not_working
	_local (scout, missing_string, unsaved?) <<
		_self.analyse_accessibility(dataset_infos)
# <= = = = = = = = = = = = = = = = = = = = = = = = = = = =

	#
	# Now raise initial warnings if necessary
	#
	_if missing_string.size > 0
	_then
		condition.raise(:swg_dsn_missing_dataset,
				:datasets,missing_string)
	_endif


	_if relocating_scheme_id _isnt _unset
	_then
	# SCHEME_PATH specifies a temporary scheme being used to
	# relocate the scheme with ID !relocating_scheme_id! - set
	# the dynamic !relocating_scheme_id! to the value specified and
	# set the dynamic !relocating_temp_scheme_id! to the ID of the
	# tempory scheme, as given by scheme_path.last
		!relocating_scheme_id! << relocating_scheme_id
		!relocating_temp_scheme_id! << scheme_path.last
	_endif 
	
	# Investigate the alternatives and check sub-tree.
	recursively? << recursively?.default(_false)
	# Note that the following method will modify most entries in scout
	# changing structure to property_list[symbol] of (boolean /
	# rope of (_unset/string)): where the alternative to be removed
	# was found the entry will be replaced by the absolute alternative
	# path.
	_self.check_alternative_occupation_and_sub_tree(scout, scheme_path, recursively?)
# <= = = = = = = = = = = = = = = = = = = = = = = = = = = = = = = = = = = = = = = = = = = =

	#
	# Start the effective task
	#

	dataset_names        << _self.dataset_names_for(dataset_infos)
	non_dc_dataset_names << _self.dataset_names_for(non_dc_managed_dsinfos)
		
	# For a top scheme reclaim the alternatives to the pool manager
	# for the datasets that are not being managed by Design Clerk

	returned_to_free_pool? << _if top_scheme? _andif _not non_dc_dataset_names.empty?
				  _then
				  # For all datasets that are not being managed by Design Clerk,
				  # return the alternatives to the pool manager as free
				  # alternatives for re-use by subsequently created new top-level
				  # designs.
					  >> swg_dsn_alt_pool_manager.free_design_alternatives(scheme_path[1], non_dc_dataset_names)
				  _else
				  # Either a sub-design or a design in a DC managed dataset - the
				  # alternative (or DC design) is therefore to be deleted. 
					  >> _false   
				  _endif

	_for ds_name _over dataset_names.fast_elements()
	_loop
		# scout[ds_name] can either be a Boolean or an alternative path
		_if (alt_path << scout[ds_name]).is_class_of?(rope)
		_then
			# It is an alternative path, thus the alternative exists and
			# must be removed now.
			ds_view << gis_program_manager.cached_dataset(ds_name)

			drop_info[ds_name] << _maybe
			
			temp_repl << ds_view.replicate_without_collections(0)
			_protect
				_local drop_env_catch << "DropEnv"
				_catch drop_env_catch
					_handling ds_alternative_cannot_be_removed, db_error _with
					_proc (cond)
						_import drop_env_catch, ds_view
						condition.raise(:swg_dsn_information,
								:string, write_string(ds_view,
										      ": ",
										      cond.report_contents_string))
						_throw drop_env_catch
				# <==================================
					_endproc

					_if returned_to_free_pool? _andif non_dc_dataset_names.includes?(ds_name)
					_then
					# Top level design altrnatatives were returned to the free pool - remove any
					# child sub-trees for a non-DC managed dataset.
						temp_repl.go_to_alternative(alt_path,:readonly)
						_for an_alt _over temp_repl.alternatives.fast_elements()
						_loop 
							temp_repl.remove_alternative(an_alt.alternative_name.write_string, _true)
						_endloop
					# go up to the parent of the target alternative.
						temp_repl.up()
					_else 
					# Go to the parent of the target alternative.

						temp_repl.go_to_alternative(alt_path.subseq(1,alt_path.size-1),:readonly)

						# Remove the whole sub-tree
						temp_repl.remove_alternative(alt_path.last,_true)
					_endif 
					# Remove the base checkpoint
					cp_name << write_string( %{, alt_path.last, %} )
					_if temp_repl.has_checkpoint?(cp_name)
					_then
						temp_repl.remove_checkpoint(cp_name)
					_endif

					# If we come hitherto then all has gone well.
					drop_info[ds_name] << _true
					
				_endcatch #drop_env_catch
			_protection
				temp_repl.discard()
			_endprotect
		_else # (alt_path << scout[ds_name]) must be a Boolean value
			# If alt_path is _false then the dataset has not been found,
			# otherwise the alternative has not existed. So its value is
			# just what is to be returned.
			drop_info[ds_name] << alt_path
		_endif # scout[ds_name]
	_endloop # for ds_name _over dataset_names.fast_elements()

	>> drop_info 

_endmethod
$

#------------------------------------------------------------------------------
# Method: available_environment_of_scheme()  KGU
#------------------------------------------------------------------------------
_pragma(classify_level=restricted,
	topic={swg, swg_design, dsn_admin, dsn_user_alt_ctrl})
_method swg_dsn_user_alternative_control.available_environment_of_scheme(scheme_path)
	## D e s c r i p t i o n :
	## Detects,in which application datasets known to the planning
	## system there are database versions associated to the scheme
	## determined by the given SCHEME_PATH. Therefore yields a
	## property list of Kleenean values, one per any dataset
	## information record stored in the swg_dsn_dataset_info table
	## of the Design Administration dataset, where the key is the
	## (internal) dataset name. 
	## 
	## A r g u m e n t s  (- <name>: <class>; <description>):
	## - SCHEME_PATH: list of sys_id; Sequence of swg_dsn_scheme record keys,
	##     representing the design path down from the related top scheme
	##     to the very scheme the investigation is to be performed for;
	##     a rope with first element 0, however, designates a
	##     build-plan pool alternative.
	##
	## D y n a m i c   V a r i a b l e s  (- <name>: <class>; <description>):
	## none.
	## 
	## R e s u l t s  (- <name>: <class>; <description>):
	## - AVAILABILITY is a look-up table, keyed by dataset names,
	##     the Kleenean entries of which having the following meening: 
	##     _true stands for an existing associated alternative in the
	##           respective dataset,
	##     _false for a definitely absent alternative in a found dataset,
	##           and
	##     _maybe for lacking information due to a missing dataset.
	## 
	## S i d e   E f f e c t s   (incl. important slot changes):
	## Temporary replicates are set up and discarded for investigation
	##   purposes.
	## 
	## C o n d i t i o n s  (- <name>: <type>; <indication>):
	## - swg_dsn_design_system_not_working: swg_dsn_error; if the design
	##     system has not been started or was switched off.

	_local ds_name                  # symbol; an (internal) dataset name
	_local path                     # rope of (_unset/string); an alternative path
	_local ds_view                  # gis_ds_view; a dataset
	
	# Check the existence and accessibility of the required datasets
	# - availability: property_list[symbol] of boolean; premature result
	#     (see header), for now, _false means that the dataset is missing;
	# May cause an error condition swd_dsn_design_system_not_working
	_local availability <<
		_self.analyse_accessibility( .dataset_infos )
# <= = = = = = = = = = = = = = = = = = = = = = = = = = = = =(swg_dsn_design_system_not_working)
	#
	# Start the effective task
	#
	_for ds_info _over .dataset_infos.fast_elements()
	_loop
		ds_name << ds_info.dataset_name.as_symbol()
		_if availability[ds_name] _is _true 
		_then
			# Check whether the path exists
			path << _self.alt_path_from_scheme_path(ds_info,scheme_path)
			path.add_first(_unset)
			ds_view << gis_program_manager.cached_dataset(ds_name)
			availability[ds_name] << _self.path_exists?(ds_view, path)
		_else
			# as dataset is not available, we can not determine wether path
			# exists or not 
			availability[ds_name] << _maybe 
		_endif # availability[ds_name] _is _true
	_endloop # for ds_info _over .dataset_infos.fast_elements()

	>> availability
_endmethod
$

#------------------------------------------------------------------------------
# Method: active_scheme_environment()
#------------------------------------------------------------------------------
_pragma(classify_level=restricted,
	topic={swg, swg_design, dsn_admin, dsn_user_alt_ctrl})
_method swg_dsn_user_alternative_control.active_scheme_environment(dataset_infos)
	## D e s c r i p t i o n :
	## For every dataset described by a DATASET_INFOS' element,
	## detects the current alternative and the corresponding scheme
	## path as well as the current access mode and returns this
	## information in property_list ACTIVE_PATHS as described below.
	## In addition the compatibility of the found scheme paths is
	## examined and returned as a Kleenean value COHERENT??.
	## 
	## A r g u m e n t s  (- <name>: <class>; <description>):
	## - DATASET_INFOS: collection of swg_dsn_dataset_info;
	##     the dataset information records for the datasets to be
	##     examined;
	## 
	## D y n a m i c   V a r i a b l e s  (- <name>: <class>; <description>):
	## 
	## R e s u l t s  (- <name>: <class>; <description>):
	## - ACTIVE_PATHS: property_list[symbol] of {list of integer, symbol};
	##     associates a vector of to values to each dataset name
	##     given by a DATASET_INFOS record:
	##     1. either a scheme path if the current alternative of
	##        the dataset is a design alternative or _unset if the
	##        dataset is not initialised or found or if the
	##        alternative cannot be translated to a scheem path;
	##     2. a symbol designating the current access mode:
	##        :none if there is no access (e.g. dataset missing),
	##        :write, :readonly, or :hard_readonly otherwise
	## - COHERENT??: kleenean; is _true if for all inquired
	##     datasets there are design alternatives and these are
	##     compatible; it is _false, if at least one dataset has an
	##     incompatible path, and it is _maybe if some datasets are
	##     not available but in the remaining datasets there are
	##     compatible paths.
	##     Note that a set of scheme paths is regarded as compatible
	##     if all of them are equal or are a prefix of the longest
	##     one of them, provided that shorter paths do not have
	##     :write mode, e.g. a set {[17,45,95,123], [17,45,95],
	##     [17,45,95,123], [17,45]} is compatible is this sense
	##     if paths [17,45] and [17,45,95] are readonly.
	## 
	## S i d e   E f f e c t s   (incl. important slot changes):
	## temporary replicates without collections are created and
	## discarded
	## 
	## C o n d i t i o n s  (- <name>: <type>; <indication>):
	## - swg_dsn_design_system_not_working: swg_dsn_error; if the design
	##     system has not been started or was switched off.

	_local active_paths << property_list.new(dataset_infos.size) # the first result
	_local coherent?? << _true      # kleenean; the second result (optimistic)
	
	_local ds_name                  # symbol; an (internal) dataset name
	_local ds_view                  # gis_ds_view; an application dataset
	_local scheme_path              # list of integer; an intermediate result
	_local access_mode              # symbol;          an intermediate result
	_local longest_path             # list of integer; the longest scheme path
	_local max_size << -1           # integer;         length of longest_path

	# Check the existence and accessibility of the required datasets
	# - availability: property_list[symbol] of boolean; premature result
	#     (see header), for now, _false means that the dataset is missing;
	# May cause an error condition swd_dsn_design_system_not_working
	_local availability <<
		_self.analyse_accessibility( dataset_infos )
# <= = = = = = = = = = = = = = = = = = = = = = = = = = = = =
	#
	# Start the effective task
	#

# OLD VERSION (tested)
	_for ds_info _over dataset_infos.fast_elements()
	_loop
		access_mode << :none
		scheme_path << _unset
		
		ds_name << ds_info.dataset_name.as_symbol()
		
		_if availability[ds_name] _is _true
		_then
			ds_view << gis_program_manager.cached_dataset(ds_name)
			
			scheme_path << _self.scheme_path_from_alt_path(
					       ds_info,
					       ds_view.alternative_path)
			access_mode << ds_view.mode
			_if scheme_path _isnt _unset _andif
			    scheme_path.size > max_size
			_then
				longest_path << scheme_path
				max_size << longest_path.size
			_endif
		_else
			# Since at least one dataset is missing the result cannot be
			# better than _maybe.
			coherent?? << _maybe
		_endif # availability[ds_name] _is _true
		
		active_paths[ds_name] << {scheme_path, access_mode}
	_endloop # for ds_info _over .dataset_infos.fast_elements()

	# NEW VERSION (draft)
	# Take advantage of the new information in .view_status
#	_for ds_info _over dataset_infos.fast_elements()
#	_loop
#		ds_name << ds_info.dataset_name.as_symbol()
#		_if (ds_view << gis_program_manager.cached_dataset(ds_name)) _is _unset _orif
#		    (vs_entry << .view_status[ds_name]) _is _unset
#		_then
#			active_paths[ds_name] << {_unset,:none}
#			coherent?? << _maybe
#		_else
#			active_paths[ds_name] << {vs_entry.scheme_path,ds_view.mode}
#		_endif
#	_endloop

	# Now, having all interesting scheme paths together, look for
	# compatibility.
	# If the effective scheme path over all configured datasets is
	# _unset then there is some incoherent version, but possibly
	# not among the requested datasets, hence we must examine the
	# requested scheme paths in this case. 
	_if .effective_scheme_path _is _unset
	_then 
		_for path_info _over active_paths.fast_elements()
		_loop
			_if (scheme_path << path_info[1]) _isnt _unset _andif
			    ((scheme_path.size < max_size _andif path_info[2] _is :write) _orif  
			     _not scheme_path.eq?( longest_path.subseq(1,scheme_path.size) ))
			_then
				coherent?? << _false
				_leave
		# <===================
			_endif
		_endloop
	_endif 

	>> active_paths, coherent??
	
_endmethod
$

#------------------------------------------------------------------------------
# Method: scheme_environment_out_of_date??()
#------------------------------------------------------------------------------
_pragma(classify_level=restricted,
	topic={swg, swg_design, dsn_admin, dsn_user_alt_ctrl})
_method swg_dsn_user_alternative_control.scheme_environment_out_of_date??(scheme_path,dataset_infos,
																		  _optional dep_scheme_paths)
	## D e s c r i p t i o n :
	## Examines in all datasets the description of which is given by
	## an swg_dsn_datset_info record of DATASET_INFOS whether there
	## is a database version associated to the scheme, characterised
	## by SCHEME_PATH that needs merging from the parent version,
	## because the latter contains changes since spawning or last
	## merge. Returns a Kleenean value which is _true if so, _false,
	## if definitely not, or _maybe if none of the accessible
	## datasets need updating but there are missing datasets.
	## If DEP_SCHEME_PATHS is given, then the result also reflects
	## the need to merge the dependent schemes characterized by the
	## paths in DEP_SCHEME_PATHS
	## 
	## A r g u m e n t s  (- <name>: <class>; <description>):
	## - SCHEME_PATH: list of sys_id; Key sequence representing the
	##     design tree path from the root (base scheme) down to the
	##     very scheme the versions are to be compared for; if
	##     SCHEME_PATH is empty, then the differences between
	##     as-built data and design-top alternative are checked;
	##     a rope with first element 0, however, designates a build-
	##     plan pool alternative;
	## - DATASET_INFOS: collection of swg_dsn_dataset_info; a set
	##     of dataset information records specifying the datasets
	##     the versions of which are to be examined.
	## - DEP_SCHEME_PATHS: list of scheme paths (see above for a
	##     description of a scheme path)
	## 
	## D y n a m i c   V a r i a b l e s  (- <name>: <class>; <description>):
	## none
	## 
	## R e s u l t s  (- <name>: <class>; <description>):
	## - OUT_OF_DATE??: kleenean;
	##     _true, if---with regard to at least one of the requested
	##        datasets---there is a need for merging a changed
	##        parent version,
	##     _false, if for definitely none of the described datasets
	##        a merging from the direct parent is needed, and
	##     _maybe, if at least one of the datasets is not available,
	##        but none of the found versions needs merging.
	## 
	## S i d e   E f f e c t s   (incl. important slot changes):
	## Two temporary replicates are needed. 
	## 
	## C o n d i t i o n s  (- <name>: <type>; <indication>):
	## swg_dsn_design_system_not_working: swg_dsn_error; if the
	##   method is called without previous announcement of the
	##   design assistant.

	_local out_of_date?? << _false # kleenean; for return value
	_local alt_path            # rope of (_unset/string); an alternative path
	_local ds_name             # symbol; an (internal) dataset name
	_local ds_view             # gis_ds_view; an application dataset
	_local repl1,repl2         # database_view; a temporary replicate

	# Check the existence and accessibility of the required datasets
	# - found?: property_list[symbol] of boolean; _false means that
	#     the dataset, the name of which serves as key, is missing;
	# May cause an error condition swd_dsn_design_system_not_working
	_local found? <<
		_self.analyse_accessibility( dataset_infos )
# <= = = = = = = = = = = = = = = = = = = = = = = = = = = = =

	all_scheme_paths << rope.new_with(scheme_path, _scatter dep_scheme_paths.default({}))

	_for ds_info _over dataset_infos.fast_elements()
	_loop @ds_info
		# We can immediately skip the dataset in following cases:
		# - the dataset has not been found;
		# - there is no version associated to the scheme beneath the
		#   wanted reference version.
		ds_name << ds_info.dataset_name.as_symbol()
		_if _not found?[ds_name]
		_then
			out_of_date?? << _maybe
		_else 
			ds_view << gis_program_manager.cached_dataset(ds_name)

			_for idx, a_scheme_path _over all_scheme_paths.fast_keys_and_elements()
			_loop
				# Find the deepest existing alternative corresponding to
				# the wanted path.
				alt_path << _self.alt_path_from_scheme_path(ds_info,a_scheme_path)
				alt_path.add_first(_unset)

				repl1 << ds_view.replicate_without_collections(0)
				_protect
					# If the scheme doesn't have an alternative within this dataset
					# then nothing must be done.
					_if _self.go_to_path_if_possible(repl1, alt_path, :readonly)
					_then
						# Now repl1 is at the scheme alternative

						_protect
							_if idx ~= 1
							_then
								# A dependent scheme - compare the virtual base and the current
								# version of the dependent scheme alternative
								vbase << _self.virtual_base_checkpoint_name_for(scheme_path.last)
								_if repl1.has_checkpoint?(vbase)
								_then
									repl2 << repl1.replicate_without_collections(0)
									repl2.go_to_checkpoint(vbase, :readonly)
								_else
									repl2 << repl1.replicate_without_collections(:base)
								_endif

							_elif _not a_scheme_path.empty? _andif
								  _not a_scheme_path.eq?({0})
							_then
								# Project or build-plan pool - compare the current version of
								# A_SCHEME with the current version of its parent alternative
								repl2 << repl1.replicate_without_collections(:base)
								repl1.up()
								repl1.rollforward()

							_else
								# A_SCHEME - compare the base version of A_SCHEME with the
								# current version of its parent alternative
								repl2 << repl1.replicate_without_collections(:base)
								repl1.up()
								repl1.rollforward()
							_endif

							out_of_date?? _or<< repl1.table_changes_to?(repl2)
						_protection
							repl2.discard()
						_endprotect
					_endif
				_protection
					repl1.discard()							
				_endprotect
				
				_if out_of_date?? _is _true
				_then
					_leave @ds_info
				_endif
			_endloop
		_endif # _not found?[ds_name]
	_endloop @ds_info

	>> out_of_date??
_endmethod
$


#------------------------------------------------------------------------------
# Iterator Method: records_modified_in_scheme_environment()
#------------------------------------------------------------------------------
_pragma(classify_level=restricted,
	topic={swg, swg_design, dsn_admin, dsn_user_alt_ctrl})
_pragma(classify_level=restricted, topic={swg, swg_design, dsn_admin, dsn_user_alt_ctrl})
_iter _method swg_dsn_user_alternative_control.records_modified_in_scheme_environment(
		      _optional scheme_path, reference, dataset_infos, only_rwos?, table_names, external_only?)
		      
	## D e s c r i p t i o n :
	## For the planning scheme referred to by SCHEME_PATH and the
	## datsaets characterised by the dataset information records
	## contained in DATASET_INFOS, iteratively calls the applied loop
	## body with the parameters described below, thus yielding all
	## database records created, modified or removed with respect
	## to the REFERENCE scheme.
	## 
	## A r g u m e n t s  (- <name>: <class>; <description>):
	## - SCHEME_PATH: list of sys_id; Sequence of swg_dsn_scheme keys,
	##     representing the scheme path down from the root (top scheme)
	##     to the very scheme. A rope with first element 0, however, designates the
	##     build-plan pool alternative. Default is the active
	##     scheme path (current environment), in case of differing
	##     paths the longest of them must cover all and is assumed
	##     to represent the very scheme---otherwise a condition will
	##     be raised; 
	## - REFERENCE: symbol/integer; either one of the symbols :this
	##     and :as_built_data or a non-negative integer less than
	##     or equal to the SCHEME_PATH length. :this means the
	##     starting version of the scheme referred to by SCHEME_PATH,
	##     :as_built_data refers to the as-built data, an integer
	##     number n refers to the n-th level above in the path;
	## - DATASET_INFOS: collection of swg_dsn_dataset_info; The dataset
	##     information records for those datasets, within which the associated
	##     alternatives are to be checked for modified objects;
	##     Default is the entire swg_dsn_dataset_info collection.
	## - ONLY_RWOS?: boolean; If this value is _true, then only RWO tables
	##     are investigated; if TABLE_NAMES is also given, then this
	##     constraint is additionally valid (i.e. only those RWO
	##     tables are examined which are named in TABLE_NAMES);
	##     Default ist _false;
	## - TABLE_NAMES: set of symbol; a set of (internal) names of
	##     interesting datastore collections; if given, then only
	##     the named tables are considered; Note: ONLY_RWOS? may
	##     interfere.
	## - EXTERNAL_ONLY?: boolean; If this value is _true, then only
	##    external tables are investigated.
	##     Default ist _false;
	## 
	## D y n a m i c   V a r i a b l e s  (- <name>: <class>; <description>):
	## none
	##
	## I t e r a t i o n   P a r a m e t e r s  (- <name>: <class>; <description>):
	## - DATASET_NAME: symbol; the internal name of the respective dataset;
	## - CHANGE: symbol; one of the values :insert, :update, :remove,
	##     denoting that the accompanying object was created,
	##     modified or removed, respectively, with respect to the
	##     reference version.;
	## - RECORD: ds_record; in case CHANGE = :insert it is a valid record,
	##     with :update and :delete, however, a detached record (in
	##     its original shape i.e. as it was in the REFERENCE version;
	##     the availability of the according source_view, however, is
	##     only granted within the iteration cycle.
	## 
	## R e s u l t s  (- <name>: <class>; <description>):
	## none
	## 
	## S i d e   E f f e c t s   (incl. important slot changes):
	## Temporary replicates are created and discarded thereafter.
	## 
	## C o n d i t i o n s  (- <name>: <type>; <indication>):
	## - swg_dsn_missing_dataset: swg_dsn_warning; if a dataset
	##     required by DATASET_INFOS cannot be found;
	## - swg_dsn_ambiguous_scheme_environment: swg_dsn_error; if SCHEME_PATH
	##     is not given but the current alternatives of the requested datasets
	##     are incompatible.

	_local default_environment? <<
		scheme_path _is _unset   # boolean; _true if the active environment is wanted
	_local changes                   # vector of ds_table_change
	_local tab_name                  # symbol; table name
	_local reference_path << {}      # list of integer; the according scheme path
	                                 #   of the reference version;
	_local comp_views                # vector(2) of {database_view,boolean}; contains the
	                                 #   comparation views (disk version scheme / reference
					 #   version) and a tag whether they are to be discarded
					 #   at the end of the loop;
	_local active_paths              # property_list[symbol] of {rope of integer, symbol};
	                                 #   per dataset name a scheme path and an access mode
	_local a_path                    # list of integer; a scheme path
	_local path_depth                # integer; the length of a scheme path rope.
	_local compatible??              # kleenean; informs whether the dataset environments
	                                 #   match.
	_local repl_sch?, repl_ref?      # boolean; _true if there is a particular replicate for
	                                 # the scheme or the reference version, respectively
	_local ds_name                   # symbol; an (internal) dataset name
	_local ds_view                   # gis_ds_view; an application dataset
	_local upper, lower              # rope of string; alternative path segments
	_local repl1,repl2               # database_view; a temporary replicate
	_local scheme_view, reference_view # database_view; the views to be compared.
	_local this_tab                  # ds_collection; a modified collection
	_local diff_str                  # ds_difference_stream
	_local this_rec, reference_rec   # ds_record/detached_record

	# Check the existence and accessibility of the required datasets
	# - found?: property_list[symbol] of boolean; dataset seek results:
	#     _false means that the dataset is missing;
	# - missing_string: character_vector; names of missing datasets;
	# May cause an error condition swd_dsn_design_system_not_working
	_local (found?, missing_string) << _self.analyse_accessibility(
					    dataset_infos.default( .dataset_infos ))
# <= = = = = = = = = = = = = = = = = = = = = = = = = = = = = = = = = = = = = = = =
	#
	# Now raise initial warnings if necessary
	#

	scheme_rel_table_names << set.new_with(:swg_dsn_scheme_relationship, :swg_dsn_sr_key_string)

	_if missing_string.size > 0 _andif
	    dataset_infos _isnt _unset
	_then
		condition.raise(:swg_dsn_missing_dataset,
				:datasets, missing_string)
	_endif

	# Completion of arguments
	dataset_infos << dataset_infos.default( .dataset_infos )
	only_rwos? << only_rwos? _is _true
	external_only? << external_only?.default(_true)

	# Detect the active environment
	(active_paths, compatible??) << _self.active_scheme_environment(dataset_infos)
	# If SCHEME_PATH was not given then examine the active environment
	_if default_environment?
	_then
		_if compatible?? _is _false
		_then
			# if the current paths don't match then it is an error
			condition.raise(:swg_dsn_ambiguous_scheme_environment)
# <===========================================================================
		_endif
		
		# Otherwise identify the longest path
		scheme_path << {}
		_for path_info _over active_paths.fast_elements()
		_loop
			_if (a_path << path_info[1]) _isnt _unset _andif
			    a_path.size > scheme_path.size
			_then
				scheme_path << a_path
			_endif
		_endloop
	_endif
	# Now, SCHEME_PATH is the path of the wanted scheme.

	_if reference _is _unset _orif
	    reference = 0
	_then
		reference << :this
	_elif reference.class_name _is :integer _andif 
	      reference > scheme_path.size
	_then
		reference << :as_built_data
	_endif
	# The reference path is set according to the given or completed REFERENCE
	reference_path << _self.determine_reference_path(reference, scheme_path)

	#
	# Now loop over the datasets and yield the wanted information
	#

	_for ds_info _over dataset_infos.fast_elements()
	_loop
		# We can immediately skip the dataset in following cases:
		# - the dataset has not been found;
		# - there is no version associated to the scheme beneath the
		#   wanted reference version.
		ds_name << ds_info.dataset_name.as_symbol()
		_if _not found?[ds_name]
		_then
			_continue
	# ^----------------------
		_endif

		ds_view << gis_program_manager.cached_dataset(ds_name)

		_if default_environment?
		_then
			path_depth << active_paths[ds_name].size
		_else
			# Find the deepest existing alternative corresponding to the
			# wanted path and split the path in an upper and a lower part,
			# where the upper part is expected to be an alternative path
			# and the lower part the remaining scheme path.
			# (The upper part should at least contain the design-top path.)
			scheme_view << ds_view.replicate_without_collections(0)
			_protect
				(upper,lower) << _self.go_as_deep_as_possible(
							ds_info, scheme_path, scheme_view) 
			_protection
				scheme_view.discard()
			_endprotect
			path_depth << scheme_path.size - lower.size
		_endif
		_if path_depth < reference_path.size
		_then
			_continue
	# ^----------------------
		_endif

		# Decide the need for replicates:
		# A replicate for the scheme's version is necessary if
		# - the scheme is not the active one.
		# A replicate for the reference version is necessary if
		# - the scheme is not the active one or
		# - the scheme is the active one but REFERENCE is not :this, or
		#                                    the base replicate is not set(?).
		repl_sch? << repl_ref? << _false
		reference_view << _unset
		_if default_environment? _orif
		    ((ap << active_paths[ds_name][1]) _isnt _unset _andif ap.eq?(scheme_path))
		_then
			# By default or chance the active scheme environment is to be
			# examined
			scheme_view << ds_view
			_if reference _isnt :this _orif
			    (reference_view << .view_status[ds_name].base_replicate_view) _is _unset
			_then
				repl_ref? << _true
			_endif
		_else
			# Another environment is to be examined.
			repl_sch? << _true
			scheme_view << ds_view.replicate(0)
			repl_ref? << _true
		_endif
		#
		# Now do the effective work
		#
		_protect
			# Adjust versions of the views to be compared
			_if repl_sch?
			_then
				scheme_view.go_to_alternative(upper, :readonly)
				# Ensure up-to-date version (beware of eventual history)
				scheme_view.rollforward()
			_endif
			_if repl_ref?
			_then
				reference_view << scheme_view.replicate(reference_path.size - path_depth)
				_if reference _is :this
				_then
					_self.do_move_replicate(reference_view,_unset,:base)
				_else
				# Ensure as-built replica view is up to date
					reference_view.rollforward()
				_endif
			_endif
				
			#
			# Now see whether there are changes
			#
			changes << scheme_view.table_changes_to(reference_view)
			
			_for ch _over changes.fast_elements()
			_loop
				_if ch.change _isnt :modified
				_then
					_continue
			# ^----------------------
				_endif
				
				tab_name << ch.tablename
				_if table_names _isnt _unset _andif
				    _not table_names.includes?(tab_name)
				_then
					_continue
			# ^----------------------
				_endif
				
				this_tab << scheme_view.collections[tab_name]

				# Do not report internal tables unless requested (the scheme
				# relationship datamodel tables are also considered internal
				# tables even though they are not flagged as such).
				
				_if external_only? _andif (this_tab.descriptor.internal_table? _orif
				    scheme_rel_table_names.includes?(tab_name)) _then _continue _endif
				
				_if this_tab.record_exemplar.is_rwo? _orif
				    _not only_rwos?
				_then
					diff_str << ds_difference_stream.new_on(
							    this_tab,
							    reference_view.collections[tab_name]
									 )
					_loop # diff_str.get()
						(change, this_rec, reference_rec) << diff_str.get()
						_if change _is _unset
						_then
							_leave
					# <===================
						_endif

						_loopbody(
							ds_name,
							change,
							_if change _is :insert
							_then
								>> this_rec
							_else
								>> reference_rec
							_endif
						)
					_endloop # diff_str.get()
					
				_endif
			_endloop # _for ch _over changes...
			
		_protection
			_if repl_sch?
			_then
				scheme_view.discard()
			_endif
			_if repl_ref? _andif
			    reference_view _isnt _unset 
			_then
				reference_view.discard()
			_endif
		_endprotect

	_endloop # _for ds_info _over dataset_infos...
	      
_endmethod
$

#------------------------------------------------------------------------------
# Method: ds_version_at_design_start()
#------------------------------------------------------------------------------
_pragma(classify_level=restricted,
	topic={swg, swg_design, dsn_admin, dsn_user_alt_ctrl})
_method swg_dsn_user_alternative_control.ds_version_at_design_start(dataset_key)
	## D e s c r i p t i o n :
	## Returns a database view of the dataset specified by DATASET_KEY,
	## the version of which is prior to the first changes made
	## within the active scheme. It may be either a replicate of
	## the start version or (if the dataset is not associated to
	## the active scheme) the current readable version (which
	## usually belongs to a parent scheme or the as-built data).
	##
	## Note that the returned view may be ephemeral or its version
	## may turn away with a change of active scheme environment. So,
	## intended database retrieval should not be delayed too much.
	## 
	## A r g u m e n t s  (- <name>: <class>; <description>):
	## - DATASET_KEY: sys_id/symbol/string; if it is a number then
	##     it is assumed to be the sys_id of an swg_dsn_dataset_info
	##     record, if it is a symbol then it is deemed the internal
	##     name of a dataset, else it is regarded as an external
	##     view name ('original view name').
	## 
	## D y n a m i c   V a r i a b l e s  (- <name>: <class>; <description>):
	## none
	## 
	## R e s u l t s  (- <name>: <class>; <description>):
	## - A_VIEW: database_view/_unset; if a suitable view can be
	##     detected it is returned here, otherwise _unset.
	## 
	## S i d e   E f f e c t s   (incl. important slot changes):
	## none
	## 
	## C o n d i t i o n s  (- <name>: <type>; <indication>):
	## none

	_local dataset_name      # symbol; an internal dataset name
	_local status_object     # swg_dsn_uac_view_status; an internal dataset information
	_local a_view            # database_view; an active view or a replicate

	# Attempt to find out the internal dataset name as key for .view_status
	_if dataset_key.is_class_of?(0)
	_then
		dataset_name << swg_dsn_engine.dataset_name_from_id(dataset_key)
	_elif _not dataset_key.responds_to?(:|as_symbol()|)
	_then
		_return _unset
# <===========================
	_elif _not dataset_key.is_class_of?(symbol) _orif
	      _not .view_status.includes_key?(dataset_key.as_symbol())
	_then
		# It seems to be a string describing the 'original view name'
		dataset_name << _self.identify_dataset_of_replicate_name(dataset_key)
	_else
		dataset_name << dataset_key
	_endif

	# Now look for a useful replicate.
	_if (status_object << .view_status[dataset_name]) _is _unset
	_then
		# The dataset has not been initialised
		_return _unset
# <===========================
	_elif (a_view << status_object.base_replicate_view) _is _unset
	_then
		# There is no base replicate, i.e. the dataset does not belong
		# to the active scheme, and we can return the readonly ancestor
		# version of the view.
		a_view << gis_program_manager.cached_dataset(dataset_name)
	_endif

	>> a_view
_endmethod
$

#------------------------------------------------------------------------------
# Method: request_replicate_view_for_scheme()
#------------------------------------------------------------------------------
_pragma(classify_level=advanced, topic={swg, swg_design, dsn_admin, dsn_user_alt_ctrl})
_method swg_dsn_user_alternative_control.request_replicate_view_for_scheme(scheme, dataset_name, version_selector,
									   _optional for_rendering?)
	##
	## D e s c r i p t i o n :
	## Returns a replicate view of the dataset keyed by the
	## DATASET_NAME argument for the planning scheme determined by
	## the given SCHEME and representing the version symbolised by
	## VERSION_SELECTOR which may be one of :current, :base, or
	## :last_build_plan. 
	## If the given SCHEME is not associated to a dataset with
	## (internal) name DATASET_NAME then a readable replicate
	## version related to the nearest ancestor scheme (by default
	## the as-built data) which is associated to this dataset will
	## be returned. 
	## If there is no dataset with this name at all, a condition
	## swg_dsn_missing_dataset will be raised.
	## The returned replicate view will be reserved until a
	## release_replicate_view() on the user alternative control is
	## performed for. This way, a re-using of released replicates
	## per dataset is possible. After the release having been done
	## the next request with the same dataset name will normally
	## recycle a cached replicate rather than create a new one. 
	##
	## A r g u m e n t s  (- <name>: <class> # <description>;):
	## - SCHEME: swg_dsn_scheme record
	##     a valid record responding to the methods scheme_path()
	##     and dataset_infos; 
	## - DATASET_NAME: symbol
	##     the internal name of the (application) dataset the
	##     replicate view is to be derived from; 
	## - VERSION_SELECTOR: symbol
	##     denotes the version the replicate view is forced to go to:
	##      :current   denotes the current disk version,
	##      :base      the starting version of the very scheme, and
	##      :last_build_plan the last reference version of the nearest
	##                 ancestor being a build-plan;
	## - FOR_RENDERING?: boolean
	##     to be given with _true if the replicate is to be used
	##     for rendering only, resulting in a weak occupation mode
	##     which may be overridden by a subsequent rendering request
	##     (which is usually a symptom of an abortion of the prior
	##     rendering having improperly ommitted the release);
	##     Default is _false (resulting in a hard reservation).
	##
	## D y n a m i c   V a r i a b l e s  (- <name>: <class>; <description>):
	## none
	## 
	## R e s u l t s  (- <name>: <class>; <description>):
	## - REPLICATE: database_view; a replicate view of the required
	##     dataset at the most appropriate version with respect to the
	##     argument settings. The returned view will always be in
	##     read-only mode.
	## 
	## S i d e   E f f e c t s   (incl. important slot changes):
	## A number of PCLs proportional to the alternative level may be
	## comsumed; Slot .replicates is likely to vary.
	## 
	## C o n d i t i o n s  (- <name>: <type>; <indication>):
	## - swg_dsn_missing_dataset: swg_dsn_warning; if the dataset denoted
	##     by DATASET_NAME could not be found.

	_local repl                   # database_view; the replicate to be returned
	
	_local scheme_path <<         # rope of integer; the scheme ids down from the root
		scheme.scheme_path(dataset_name)
	_local levels                 # integer; the path distance to a build plan
	_local alt_path               # rope of string; the effective alternative path
	_local ds_info                # swg_dsn_dataset_info; the dataset info record
	_local vs_entry << .view_status[dataset_name]  # swg_dsn_uac_view_status

	# First try the quickest response: If the base version of the
	# current scheme environment is requested, then simply return it.
	_if version_selector _is :base _andif
	    vs_entry _isnt _unset _andif
	    vs_entry.scheme_path.eq?(scheme_path) _andif 
	    (repl << vs_entry.base_replicate_view) _isnt _unset
	_then
		_return repl
# <=========================
	_endif

	# Unfortunately, the job is a little bit harder.
	ds_info << .dataset_infos.select(
			   predicate.eq(:dataset_name, dataset_name.as_charvec())
				  ).an_element()
	
	_if .view_status[dataset_name] _is _unset _orif
	    (ds_view << gis_program_manager.cached_dataset(dataset_name)) _is _unset _orif
	    ds_info _is _unset
	_then
		condition.raise(:swg_dsn_missing_dataset,
				:datasets, dataset_name)
# <=====================================================
	_endif

	_if version_selector _is :last_build_plan
	_then
		levels << scheme_path.size - scheme.levels_up_to_a_build_plan()

		_if levels <= 0
		_then
			scheme_path.empty()
		_else
			scheme_path << scheme_path.subseq(1,levels)
		_endif
		version_selector << :current
	_endif
	
	ok? << _false
	_protect
	_protection # NOTE: The following code block has been made non-interruptable 
		_protect 
			repl << _self.do_request_replicate(ds_view, for_rendering?)
			alt_path << _self.alt_path_from_scheme_path(ds_info,scheme_path)
			_self.do_move_replicate(repl, alt_path, version_selector, _true)
			ok? << _true 
		_protection
			_if _not ok? _andif repl _isnt _unset
			_then
				_self.do_release_replicate(repl)
				repl << _unset 
			_endif 
		_endprotect
	_endprotect 
	>> repl
_endmethod
$

#------------------------------------------------------------------------------
# Method: release_replicate_view()
#------------------------------------------------------------------------------
_pragma(classify_level=advanced,
	topic={swg, swg_design, dsn_admin, dsn_user_alt_ctrl})
_pragma(classify_level=advanced, topic={swg, swg_design, dsn_admin, dsn_user_alt_ctrl})
_method swg_dsn_user_alternative_control.release_replicate_view(replicate, _optional dataset_name)

	## D e s c r i p t i o n :
	## Releases the given replicate view for discarding or
	## recycling if it had been requested by
	## _self.request_replicate_view_for_scheme() before. By
	## providing the argument DATASET_NAME the task can be
	## facilitated. If there are more than a fix limit of cached
	## replicates for the same dataset, then the one being released
	## will be discarded and dropped from the cache, otherwise it
	## will only be marked free but kept in the cache.
	#  (The limit is defined in constant max_repl_per_dataset.)
	## The next request for the same dataset name will rather
	## recycle a cached released replicate than create a new one. 
	## 
	## A r g u m e n t s  (- <name>: <class>; <description>):
	## + REPLICATE: database_view; The replicate view to be
	##     released, this should be a replicate view created or
	##     provided by _self.request_replicate_view_for_scheme(); 
	## - DATASET_NAME: symbol; the internal name of the
	##     (application) dataset the replicate view is to be
	##     derived from; 
	## 
	## D y n a m i c   V a r i a b l e s  (- <name>: <class>; <description>):
	## none
	## 
	## R e s u l t s  (- <name>: <class>; <description>):
	## none
	## 
	## S i d e   E f f e c t s   (incl. important slot changes):
	## May increase the number of used ds_environment
	## slots. Contents of slot .replicates is likely to be
	## modified. 
	## 
	## C o n d i t i o n s  (- <name>: <type>; <indication>):
	## none

	# First try the quickest solution: If the replicate to be
	# released is one of the permanently held base versions, then
	# nothing is to be done.

	_local vs_entry << .view_status[dataset_name] # swg_dsn_uac_view_status
	
	_if vs_entry _isnt _unset
	_then
		_if vs_entry.base_replicate_view _is replicate
		_then
			_return
# <============================
		_endif
	_else
		_for vs _over .view_status.fast_elements()
		_loop
			_if vs.base_replicate_view _is replicate
			_then
				_return
# <====================================
			_endif
		_endloop
	_endif

	# The job will be harder: The replicate must be found in the
	# whole pool.
	_self.do_release_replicate(replicate)
	
_endmethod
$

#------------------------------------------------------------------------------
# Method: create_checkpoint()
#------------------------------------------------------------------------------
_pragma(classify_level=restricted,
	topic={swg, swg_design, dsn_admin, dsn_user_alt_ctrl})
_method swg_dsn_user_alternative_control.create_checkpoint(scheme_path, dataset_infos, name,
					 _optional a_time)
	## D e s c r i p t i o n :
	## Simultaneously creates checkpoints with the given name NAME
	## in the alterantives associated to the scheme characterised
	## by SCHEME_PATH in all datasets described by DATASET_INFOS.
	## 
	## A r g u m e n t s  (- <name>: <class>; <description>):
	## - SCHEME_PATH: list of sys_id; Key sequence representing the
	##     design tree path from the root (base scheme) down to the
	##     very scheme the checkpoints are to be created for;
	##     a scheme path with first element 0, however, designates
	##     a build-plan pool alternative;
	## - DATASET_INFOS: collection of swg_dsn_dataset_info; a set
	##     of dataset information records specifying the datasets
	##     checkpoints are to be created in;
	## - NAME: string; the name of the checkpoint to be set in the
	##     requested datasets simultaneously.
	## - TIME: date_time; if this is given, then a special time code
	##     will be encrypted into the checkpoint name such that its
	##     creation time will always be known (this is for consistency
	##     checks among the versions and to be able to restore the
	##     original sequence.
	## 
	## D y n a m i c   V a r i a b l e s  (- <name>: <class>; <description>):
	## none
	## 
	## R e s u l t s  (- <name>: <class>; <description>):
	## - CHANGES: property_list[symbol] of kleenean; the keys are
	##     the names of the datasets described in DATASET_INFOS,
	##     the associated entries mark whether changes were to be
	##     saved (_true) or not (_false), or if the creation of the
	##     checkpoint failed (_maybe).
	## 
	## S i d e   E f f e c t s  (incl. important slot changes):
	## none
	## 
	## C o n d i t i o n s  (- <name>: <type>; <indication>):
	## - swg_dsn_design_system_not_working: swg_dsn_error; if the design
	##     system has not been started or was switched off;
	## - swg_dsn_missing_dataset: swg_dsn_warning; if required datasets are
	##     not found;
	## - swg_dsn_missing_alternative: swg_dsn_warning; if in a
	##     requested dataset the according alternative(s) are missing;
	## - swg_dsn_checkpoint_exists: swg_dsn_warning; if in one of
	##     the target alternatives a checkpoint NAME has already
	##     existed; if ignored then the old checkpoint is replaced.

	_local ds_name         # symbol; the internal name of a dataset
	_local ds_view         # gis_ds_view: a dataset
	_local need_replicate? # boolean; flag for usage of a replicate for
	                       # setting the checkpoint.
	_local ready?          # boolean; decides whether the target alternative
	                       #   exists such that it is ready to be checkpointed
	_local alt_path        # rope of (_unset/string); an alternative path

	_local cp_name << name # string

	_local changes << property_list.new() # the result (se above)
	_local working_view    # ds_version_view; a view for moving
	
	# Check the existence and accessibility of the required datasets
	# - found?: property_list[symbol] of boolean; dataset seek results:
	#     _false means that the dataset is missing;
	# - missing_string: character_vector; names of missing datasets;
	# May cause an error condition swd_dsn_design_system_not_working
	_local (found?, missing_string) <<
		_self.analyse_accessibility(dataset_infos)
# <= = = = = = = = = = = = = = = = = = = = = = = = = = = =

	#
	# Now raise initial warnings if necessary
	#
	_if missing_string.size > 0
	_then
		condition.raise(:swg_dsn_missing_dataset,
				:datasets, missing_string)
	_endif

	#
	# Start the effective task
	#
	_for ds_info _over dataset_infos.fast_elements()
	_loop
		ds_name << ds_info.dataset_name.as_symbol()
		_if found?[ds_name]
		_then
			ds_view << gis_program_manager.cached_dataset(ds_name)
			# Decide whether we can set the checkpoint on the active version
			need_replicate? <<
				(cur_sch_path << .view_status[ds_name].scheme_path) _is _unset _orif
				_not cur_sch_path.eq?(scheme_path)
			_if need_replicate?
			_then
				working_view << ds_view.replicate_without_collections(0,_true)
			_else
				working_view << ds_view
			_endif
			_protect
				_if need_replicate?
				_then
					alt_path << _self.alt_path_from_scheme_path(ds_info,scheme_path)
					alt_path.add_first(_unset)
					ready? << _self.go_to_path_if_possible(working_view,alt_path,:readonly)
				_else
					ready? << _true
				_endif
				_if ready?
				_then
					_if working_view.has_checkpoint?(cp_name)
					_then
						# Note: If the caller aborts or throws, then we
						# will leave a partial checkpoint set!  Note that the
						# checkpoints() method on self will only return
						# checkpoints that exist in all datasets.
						condition.raise(:swg_dsn_checkpoint_exists,
								:dataset, ds_view.original_view_name,
								:checkpoint_name, name)
					_endif
					changes[ds_name] << working_view.rollforward()
					changes[ds_name] _or<< working_view.checkpoint(cp_name)
				_else
					changes[ds_name] << _maybe
					# Note: If the caller aborts or throws, then we
					# will leave a partial checkpoint set!  Note that the
					# checkpoints() method on self will only return
					# checkpoints that exist in all datasets.
					condition.raise(:swg_dsn_missing_alternative,
							:dataset, working_view.original_view_name)
				_endif
			_protection
				_if need_replicate?
				_then
					working_view.discard()
				_endif 
			_endprotect
		_endif
	_endloop

	>> changes
	
_endmethod
$

_pragma(classify_level=restricted, topic={swg_design})
_method swg_dsn_user_alternative_control.checkpoints(scheme_path, dataset_infos,
					 _optional user_only?, same_dd_only?)
	## D e s c r i p t i o n :
	## For each dataset requested by DATASET_INFOS, yields a list
	## of checkpoint names (as they are) at the version associated
	## to the scheme specified by SCHEME_PATH. If an associated
	## dataset or alternative is not accessible then an appropriate
	## warning is induced.
	## 
	## A r g u m e n t s  (- <name>: <class>; <description>):
	## - SCHEME_PATH: list of sys_id; Key sequence representing the
	##     design tree path from the root (base scheme) down to the
	##     very scheme the checkpoints are to be investigated for;
	##     a scheme path with first element 0, however, designates
	##     a build-plan pool alternative;
	## - DATASET_INFOS: collection of swg_dsn_dataset_info; a set
	##     of dataset information records specifying the datasets
	##     checkpoints are to be searched in;
	## - USER_ONLY?: if true, then only checkpoints prefaced with
	##     the string "user_" are returned.  Default is false.
	## - SAME_DD_ONLY?: if true, then only checkpoints with the same
	##     DD version are returned.  Default is false.
	##
	## D y n a m i c   V a r i a b l e s  (- <name>: <class>; <description>):
	## none
	## 
	## R e s u l t s  (- <name>: <class>; <description>):
	## - CP_INFOS: a sorted collection of all checkpoint names satisfying the input
	##     arguments, each of which must exist in all datasets.
	## 
	## S i d e   E f f e c t s   (incl. important slot changes):
	## none
	## 
	## C o n d i t i o n s  (- <name>: <type>; <indication>):
	## - swg_dsn_missing_dataset: swg_dsn_warning; if required datasets are
	##     not found;
	## - swg_dsn_missing_alternative: swg_dsn_warning; if in a
	##     requested dataset the according alternative(s) are missing;

	cp_data << property_list.new()
	repl << _unset
	
	_protect
		_for ds_info _over dataset_infos.fast_elements()
		_loop
			# If the actual dataset does not exist, then ignore this
			# ds_info record entirely.
			_if (actual_dataset << ds_info.referred_dataset) _is _unset
			_then
				_continue
			_endif
			
			# Keep track of the checkpoints associated with each ds_info separately.
			ds_name << ds_info.dataset_name.as_symbol()
			r << cp_data[ds_name] << rope.new()
			
			# Use a replicated view to traverse to the scheme alternative,
			# and analyze the list of checkpoints.
			# If the view is clerk() managed, then we need the full replica.
			repl << _if actual_dataset.clerk() _isnt _unset
				_then >> actual_dataset.replicate()
				_else >> actual_dataset.replicate_without_collections()
				_endif
			
			repl.auth_view << _unset 
			alt_path << _self.alt_path_from_scheme_path(ds_info, scheme_path)

			_if same_dd_only? _is _true
			_then
				# Open the dd!conversions table, and prepare a procedure that
				# can be used to query the DD version at each checkpoint.
				ct << actual_dataset.collections[:dd!conversions]
				dd_file << ct.source_file.name
				version_in <<
					_proc(table)
						t << table.all_at(0)
						_if t.size = 0
						_then _return 0
						_elif t.size ~= 1
						_then _return -1
						_endif
						>> t.an_element().version2
					_endproc
				
				table << repl.open_collection(
						 dd_file, :dd!conversions,
						 actual_dataset.collections[:dd!conversions].descriptor)
			_endif
			
			# Go to the leaf alternative.  If it doesn't exist, just continue
			# quietly - but note that the ultimate result is that the list
			# of checkpoints will be empty, since it is certain that no
			# checkpoint exists in all alternatives.
			ok? << _false 
			_catch :ds_error
			       _handling db_thing_not_found _with
			       _proc(cond)
				       _throw :ds_error
			       _endproc
			       repl.go_to_alternative("|")
			       repl.go_to_alternative(alt_path)
			       ok? << _true
			_endcatch
			_if ok? _is _false
			_then
				repl.discard()
				repl << _unset 
				_continue
			_endif

			_if same_dd_only? _is _true
			_then
				start_dd_version << version_in(table)
			_endif
			
			_for cp _over repl.checkpoints.fast_elements()
			_loop
				cp_name << cp.checkpoint_name

				# If only returning user checkpoints, then ensure that the
				# name starts with "user_"
				_if user_only? _is _true _andif
				    cp_name.index_of_seq("user_") ~= 1
				_then
					_continue
				_endif

				# If only returning checkpoints with the same DD version,
				# then ensure that the DD version at this checkpoint is the
				# same as the starting value.
				_if same_dd_only? _is _true
				_then
					repl.go_to_checkpoint(cp_name)
					_if start_dd_version ~= version_in(table)
					_then
						_continue
					_endif 
				_endif

				# At this point, all prerequisites are OK, so add it to the
				# list of checkpoints for this dataset.
				r.add_last(cp_name)
			_endloop
			repl.discard()
			repl << _unset 
		_endloop
	_protection
		_if repl _isnt _unset
		_then
			repl.discard()
			repl << _unset
		_endif
	_endprotect

	# At this point, cp_data is a property_list of checkpoints in
	# each dataset.  The results must be intersected, to produce a
	# list of checkpoints that exists in all datasets.
	res << _unset
	_for i _over cp_data.fast_elements()
	_loop
		_if res _is _unset
		_then
			res << equality_set.new_from(i)
		_else
			res << res.intersection(equality_set.new_from(i))
		_endif
	_endloop
	_return res.as_sorted_collection()
			
_endmethod
$

_pragma(classify_level=restricted, topic={swg_design})
_method swg_dsn_user_alternative_control.remove_user_checkpoints(scheme_path, dataset_infos)
	##
	## Removes all user checkpoints (prefaced by "user_") from the
	## scheme environment identified by SCHEME_PATH and DATASET_INFOS
	##
	
	_protect
		_for ds_info _over dataset_infos.fast_elements()
		_loop
			# If the actual dataset does not exist, then ignore this
			# ds_info record entirely.
			_if (actual_dataset << ds_info.referred_dataset) _is _unset
			_then
				_continue
			_endif
			
			# Use a replicated view to traverse to the scheme alternative,
			# and analyze the list of checkpoints.
			repl << actual_dataset.replicate_without_collections()
			repl.auth_view << _unset 
			alt_path << _self.alt_path_from_scheme_path(ds_info, scheme_path)
			
			# Go to the leaf alternative.  If it doesn't exist, just continue
			# quietly.
			ok? << _false 
			_catch :ds_error
			       _handling db_thing_not_found _with
			       _proc(cond)
				       _throw :ds_error
			       _endproc
			       repl.go_to_alternative("|")
			       repl.go_to_alternative(alt_path)
			       ok? << _true
			_endcatch
			_if ok? _is _false
			_then
				repl.discard()
				repl << _unset 
				_continue
 			_endif

			_for cp _over repl.checkpoints.fast_elements()
			_loop
				cp_name << cp.checkpoint_name
				_if cp_name.index_of_seq("user_") ~= 1
				_then
					_continue
				_endif
				repl.remove_checkpoint(cp_name)
			_endloop
			repl.discard()
			repl << _unset 
		_endloop
	_protection
		_if repl _isnt _unset
		_then
			repl.discard()
			repl << _unset
		_endif
	_endprotect

_endmethod

#------------------------------------------------------------------------------
# Method: go_to_checkpoint()
#------------------------------------------------------------------------------
_pragma(classify_level=restricted,
	topic={swg, swg_design, dsn_admin, dsn_user_alt_ctrl})
_method swg_dsn_user_alternative_control.go_to_checkpoint(scheme_path, dataset_infos, name,
					 _optional wanted_mode, discard_checkpoints?)
	## D e s c r i p t i o n :
	## The aim is, simultaneously to go to the checkpoints with
	## given NAME in all datasets specified by dataset information
	## records in DATASET_INFOS.
	## Therefore in all datasets is checked, whether the given
	## SCHEME_PATH corresponds to the current alternative or
	## not. If not then a condition is raised. Otherwise the wanted
	## checkpoint is gone to. The NAME of the checkpoint must be exact.
	## In general, all raised warn ngs may be ignored. The
	## consistency of the scheme environment, however, is likely to
	## be corrupted. If in an associated dataset the wanted
	## checkpoint is not found then the dataset is switched to
	## readonly mode to avoid manipulations of wrong versions. The
	## same holds if in some dataset the requested checkpoint belongs
	## to a different data model version: an obsolete data model
	## version can never be made to a writable active version.
	#
	# Design comment (KGU):
	# If we handed over the time stamp management from the admin
	# engine to the user alternative control than we would be able
	# to check whether an alternative is younger than a requested
	# checkpoint and so to make the starting version available for
	# any older checkpoint. The only necessary addition would be to
	# add an initial checkpoint because otherwise we couldn't
	# return there.
	##
	## A r g u m e n t s  (- <name>: <class>; <description>):
	## - SCHEME_PATH: list of sys_id; Key sequence representing the
	##     design tree path from the root (base scheme) down to the
	##     very scheme the checkpoint is to be seeked in;
	##     a scheme path with first element 0, however, designates
	##     a build-plan pool alternative;
	## - DATASET_INFOS: collection of swg_dsn_dataset_info; the dataset
	##     information records describing the interesting datasets;
	## - NAME: string; The (exact) name of the requested checkpoint;
	## - WANTED_MODE: symbol; one of the keys :write, :readonly, or
	##     :hard_readonly (where :hard_readonly is treated like
	##     :readonly); Default is :readonly; 
	## - DISCARD_CHECKPOINTS?: boolean; it is ignored, if
	##     WANTED_MODE isn't :write, otherwise value  _true means, that
	##     all checkpoints situated between the current disk version
	##     and the given checkpoints NAME will irreversibly
	##     deleted. This option is rejected, however, if the checkpoint
	##     sequence differs among the datasets or if checkpoint NAME is
	##     missing in some datasets. Default is _false.
	## 
	## D y n a m i c   V a r i a b l e s  (- <name>: <class>; <description>):
	## none
	## 
	## R e s u l t s  (- <name>: <class>; <description>):
	## - STATUS: property_list[symbol] of {kleenean, symbol}; the
	##     keys are the names of all datasets an swg_dsn_dataset_info
	##     record was given in DATASET_INFOS; the correspondig entries
	##     are vectors the first element of which describes whether
	##     there was a difference to the former version (_true) or not
	##     (_false) or if e.g. because of a missing dataset no
	##     information is available, while the second element denotes
	##     the achieved access mode (:readonly, :hard_readonly or
	##     :write). The access mode never exceeds the WANTED_MODE and
	##     will never be :write, if the checkpoint or the dataset was
	##     missing or belong to an ancient data model
	## 
	## S i d e   E f f e c t s   (incl. important slot changes):
	## 
	## C o n d i t i o n s  (- <name>: <type>; <indication>):
	## - swg_dsn_ambiguous_scheme_environment: swg_dsn_error; if
	##     SCHEME_PATH does not match with the current alternatives
	##     of the requested datasets (or these are incompatible); 
	## - swg_dsn_missing_dataset: swg_dsn_warning; if some of the
	##     requested datasets are missing;
	## - swg_dsn_missing_checkpoint: swg_dsn_warning; if in some
	##     of the datasets a checkpoint NAME doesn't exist. 


        remove_checkpoints? << discard_checkpoints?.default(_self.dsm_discard_checkpoints?)
	
	gpm << gis_program_manager

	checkpoint_list << _self.checkpoints(scheme_path, dataset_infos, _true, _true)
	
	# Be sure checkpoint name is in Scheme's checkpoint_list
	_if _not checkpoint_list.includes?( name )
	_then
		# Raise a user error condition
		condition.raise(:swg_dsn_checkpoint_not_found_in_scheme)
	_endif

	_protect
		ok? << _false 
		# Go to the checkpoint in each dataset (in write mode)
		_for ds_infos _over dataset_infos.fast_elements()
		_loop
		
			dataset << gpm.cached_dataset( ds_infos.dataset_name.as_symbol() )
			_if dataset.has_checkpoint?( name )
			_then
				date_for_goto_checkpoint << _self.get_checkpoint_date(name, dataset)
				
				# Make a copy of the current checkpoint before going to the
				# wanted checkpoint in case anything goes wrong.
				#
				dataset.checkpoint("pre_go_to_checkpoint")
			_endif
		_endloop
		
	
		# Go to the checkpoint in each dataset (in write mode)
		_for ds_infos _over dataset_infos.fast_elements()
		_loop
			dataset << gpm.cached_dataset( ds_infos.dataset_name.as_symbol() )
			_if dataset.has_checkpoint?( name )
			_then
				#
				# Go to the wanted checkpoint
				#
				dataset.go_to_checkpoint(name, :write)
				
			_endif
		_endloop
		
		# Go to the checkpoint in each dataset (in write mode)
		_for ds_infos _over dataset_infos.fast_elements()
		_loop
			dataset << gpm.cached_dataset( ds_infos.dataset_name.as_symbol() )
			_if dataset.has_checkpoint?( name )
			_then
			        # After goto checkpoint, remove the checkpoint
				# and all more recent checkpoints
				
				_for a_cp _over dataset.checkpoints.fast_elements()
				_loop
					a_cp_name << a_cp.checkpoint_name
					
					_if date_for_goto_checkpoint _isnt _unset _andif
					    a_cp.time_created >= date_for_goto_checkpoint
					_then
						# Only remove the "user_" checkpoints
						_if a_cp_name.size > 4 _andif remove_checkpoints? _andif
						    a_cp_name.slice(1,5) = "user_"
						_then
							dataset.remove_checkpoint( a_cp_name )
						_endif
					_endif
				_endloop
			_endif
			
		_endloop
		
	ok? << _true

	#
	# If anything goes wrong, then raise a condition, go back to
	# the temp checkpoint with the name "pre_go_to_checkpoint", and
	# remove that temp checkpoint.
	#
	_protection
		
		_for ds_infos _over dataset_infos.fast_elements()
		_loop
			dataset << gpm.cached_dataset(ds_infos.dataset_name.as_symbol())
			_if dataset.has_checkpoint?("pre_go_to_checkpoint")
			_then
				_if _not ok?
				_then
					dataset.go_to_checkpoint("pre_go_to_checkpoint", :write)
				_endif
				dataset.remove_checkpoint("pre_go_to_checkpoint")
			_endif
		_endloop
		
		_if _not ok?
		_then
			condition.raise(:swg_dsn_checkpoint_not_restore)
		_endif
		
	_endprotect
	

_endmethod
$

#-----------------------------------------------------------------------------
# METHOD get_checkpoint_date()
#----------------------------------------------------------------------------
_pragma(classify_level=restricted, topic={swg, swg_design, dsn_admin, dsn_user_alt_ctrl})
_method swg_dsn_user_alternative_control.get_checkpoint_date(name, dataset)
	##
	## Get date and time of a given checkpoint from a dataset
	##
	checkpoint_date << _unset
	_for a_cp _over dataset.checkpoints.fast_elements()
	_loop
		_if (a_cp_name << a_cp.checkpoint_name) = name
		_then
			checkpoint_date << a_cp.time_created
		_endif
	_endloop
	_return checkpoint_date
_endmethod



#------------------------------------------------------------------------------
# METHOD second_alternative_name()  KGU
#------------------------------------------------------------------------------
_pragma(classify_level=advanced, topic={swg_design,dsn_user_alt_ctrl, swg_merger}, usage={redefinable})
_method swg_dsn_user_alternative_control.second_alternative_name(a_dataset_name, an_alternative_path_name)
	## D e s c r i p t i o n :
	## In the dataset with name A_DATASET_NAME searches for an
	## alternative with path name AN_ALTERNATIVE_PATH_NAME, decides
	## whether it is an alternative associated with a design scheme,
	## and if so, returns the name of the latter.
	##
	## This method is a particular interface for the merge menus
	## (swg_merge_config_menu and swg_merge_menu), arranged by a
	## broker object swg_dsn_merge_information.
	##
	## A r g u m e n t s  (- <name>: <class> # <description>;):
	## - A_DATASET_NAME: symbol # the name of the dataset the
	##     alternative belongs to;
	## - AN_ALTERNATIVE_PATH_NAME: <string> # the path as from
	##     database_view.alternative_path_name() as string. Note
	##     that the top alternative path is never supposed to be
	##     given here!
	##
	## D y n a m i c   V a r i a b l e s  (cf. Arguments):
	## None.
	##
	## R e s u l t s  (- <name>: <class> # <description>;):
	## - DISPLAY_NAME: <string>|_unset # the name of the associated
	##     scheme if the alternative was created by the design
	##     system for a scheme, otherwise it will either be the
	##     regular alternative name (if AN_ALTERNATIVE_PATH_NAME
	##     refers to an existing alternative) or _unset (else).
	##
	## S i d e   E f f e c t s   (incl. important slot changes):
	## May initialise slot .dataset_infos
	##
	## C o n d i t i o n s  (- <name>: <type>; <indication>):
	## - swg_dsn_uac_badly_initialised: swg_dsn_warning # if the
	##     dataset information cannot properly be established.

	_local display_name               # <string>|_unset; the result (see above)
	_local path <<                    # _unset|rope of (_unset|<string>)
		_self.decompose_alternative_name(an_alternative_path_name)
	_local ds_info                    # swg_dsn_dataset_info; the dataset information record
	_local ds_info_pred << predicate.eq(:dataset_name,a_dataset_name)
	_local scheme_path                # list of integer; the scheme path
	_local a_scheme                   # swg_dsn_scheme; the associated scheme record

	# path is either _unset (in case AN_ALTERNATIVE_PATH_NAME was
	# malformed) or a list of alternative names but first element _unset.
	_if path _isnt _unset _andif
	    path.size > 1
	_then
		# The default: the regular alternative name
		display_name << path[path.size]
		
		path << path.subseq(2)    # the leading _unset element is unwanted here.

		_if _self.is_design_alternative?(a_dataset_name, path)
		_then
			# Now it is proved that the alternative path points somewhere
			# into the design alternative tree.
			# Hence, the ds_info must exist.
			ds_info << .dataset_infos.select(ds_info_pred).an_element()

			# It might still be the case, though, that there is no according
			# scheme record (anymore). So better check.
			_if (scheme_path << _self.scheme_path_from_alt_path(ds_info,path)) _isnt _unset _andif
			    scheme_path.size > 0 _andif
			    (a_scheme << ds_info.source_view.collections[:swg_dsn_scheme].at(scheme_path.last)) _isnt _unset
			
			_then
				display_name << a_scheme.name
			_endif
			# (In any other case adhere to the default)
		_endif
	_endif
		
	>> display_name
	
_endmethod
$

# Method: note_change()  KGU
#------------------------------------------------------------------------------
_pragma(classify_level=advanced,
	topic={swg, swg_design, dsn_user_alt_ctrl},
	usage=subclassable)
_method swg_dsn_user_alternative_control.note_change(who_changed, what_changed,
						     _optional change_info, _gather  more_change_info)
	## D e s c r i p t i o n :
	## Registers all relevant operations, in particular on the
	## controlled application datasets and updates the .view_status
	## informations for topical environment information.
	## If WHAT_CHANGED is a controlled gis_ds_view a .view_status
	## entry exists for then its contents is updated, at least for
	## WHAT_CHANGED/CHANGE_INFO aspects :alternative_path,
	## :view/:changes, :mode
	## 
	## A r g u m e n t s  (- <name>: <class>; <description>):
	## - WHO_CHANGED: object; the notifying object;
	## - WHAT_CHANGED: symbol; the aspect key of change;
	## - CHANGE_INFO: object; an optional relevant information,
	##     depending on WHO_CHANGED and WHAT_CHANGED;
	## - MORE_CHANGE_INFO: sequence of object; further information
	## 
	## D y n a m i c   V a r i a b l e s  (- <name>: <class>; <description>):
	## none
	## 
	## R e s u l t s  (- <name>: <class>; <description>):
	## none
	## 
	## S i d e   E f f e c t s   (incl. important slot changes):
	## may change .view_status entries
	## 
	## C o n d i t i o n s  (- <name>: <type>; <indication>):
	## none

	_local status_entry                      # swg_dsn_uac_view_status
	_local ds_name                           # symbol, name of the gis_ds_view
	_local ds_info                           # swg_dsn_dataset_info
	
	
	_if who_changed.is_kind_of?(gis_ds_view) _andif
	    (status_entry << .view_status[ds_name << who_changed.name]) _isnt _unset
	_then
		_if what_changed _is :alternative_path
		_then
			status_entry.current_alternative <<
				who_changed.alternative_path.copy()
			_if .dataset_infos _isnt _unset _andif
			    (ds_info << .dataset_infos.select(predicate.eq(:dataset_name,ds_name.as_charvec())).an_element()) _isnt _unset
			_then
				status_entry.scheme_path <<
					_self.scheme_path_from_alt_path(ds_info,
									status_entry.current_alternative)
			_endif
			_self.set_effective_scheme_path()
		_endif
	_endif
	
	_super.note_change(who_changed,what_changed, change_info, _scatter  more_change_info)
	
_endmethod

#------------------------------------------------------------------------------
# Method: initialise()   KGU
#------------------------------------------------------------------------------
_pragma(classify_level=restricted,
	topic={swg, swg_design,dsn_admin})
_private _method swg_dsn_user_alternative_control.initialise()
	## D e s c r i p t i o n :
	## Initialises _self, i.e. sets up the slots, the dataset environment
	## and view status information.
	## 
	## A r g u m e n t s  (- <name>: <class>; <description>):
	## none
	## 
	## D y n a m i c   V a r i a b l e s  (- <name>: <class>; <description>):
	## none
	## 
	## R e s u l t s  (- <name>: <class>; <description>):
	## none
	## 
	## S i d e   E f f e c t s   (incl. important slot changes):
	## slots .dataset_infos, .replicates, and .view_status will be
	## initialised.
	## 
	## C o n d i t i o n s  (- <name>: <type>; <indication>):
	## - swg_dsn_design_view_not_found: swg_dsn_error; if the design
	##   administration view is not found (with WORKING? = _true only);
	## - swg_dsn_no_dataset_info_found: swg_dsn_error; if the
	##     dataset information collection can not be found or is empty;
	## - ds_view_updated: user_error; if one of the datasets
	##   relevant for planning contains unsaved changes.
	## - db_thing_not_found: error; if the configured as-built data
	##   alternative does not exist.
	## - swg_dsn_standard_version_menu_found: swg_dsn_information; if an active
	##   instance of class version_form_editor was detected.
	##

	_local ds_name         # symbol; a dataset name
	_local dataset         # database_view; a dataset to be initialised
	_local ds_info_set << equality_set.new()  # equality_set of swg_dsn_dataset_info

	# First find the dataset information collection (this may raise
	# a condition!) and set the slot .dataset_infos
	_self.set_dataset_infos()
	
	# Now investigate the dataset environment and set up the status knowledge
	# entries about the relevant datasets.
	.view_status.empty()
	_for ds_info _over .dataset_infos.fast_elements()
	_loop
		ds_name << ds_info.dataset_name.as_symbol()
		
		_if (dataset << gis_program_manager.cached_dataset(ds_name)) _is _unset
		_then
			# The configured dataset information does not correspond to an
			# initialised dataset. For now, skip the entry.
			_continue
	# ^----------------------
		_elif dataset.updated?
		_then
			# There are unsaved changes -- force a condition as it should
			# be decided what to do with before.
			# POSSIBLE IMPROVEMENT: An own condition could report the
			# name(s) of the updated views, perhaps be appropriately
			# handled.
			condition.raise(:ds_view_updated)
# <=======================================================
		_endif
		
		# Raise a condition, if the configured as-built path does not
		# exist.
		_self.check_as_built_path(ds_info,dataset)
# <= = = = = = = = = = = = = = = = = = = = = = = = = = = =
		.view_status[ds_name] << swg_dsn_uac_view_status.new(dataset)
		# Enhance the set to be passed to method go_to_scheme_environment()
		ds_info_set.add(ds_info)
		# Activate automatic updating
		dataset.add_dependent(_self, :alternative_path, :view, :mode, :commit)
	_endloop
	
	# Discard all eventually cached replicates and empty the
	# replicate cache.
	_self.remove_reference_replicates()

	# Now move all datasets to the respective planning top alternatives
	# (if these do not exist then create them)
	_self.go_to_design_tops(ds_info_set)

	_self.set_effective_scheme_path()
	
_endmethod
$

_private _method swg_dsn_user_alternative_control.set_dataset_infos(_optional quiet?)
	## D e s c r i p t i o n :
	## Initialises the slot .dataset_infos with the respective

	#
	# This method was mainly intended for testing simplification as

	# information table.
	##

	## - QUIET?: boolean; If this is _true, then in case of missing
	##     or empty dataset information table a condition
	##     swg_dsn_no_dataset_info_found will Not be raised (otherwise
	##     it will).
	## 
	## D y n a m i c   V a r i a b l e s  (- <name>: <class>; <description>):
	## none
	## 
	## R e s u l t s  (- <name>: <class>; <description>):
	## - DATASET_INFOS: ds_collection; the database collection of configured
	##     dataset information records.
	## 
	## S i d e   E f f e c t s   (incl. important slot changes):

	##

	##     Administration view is not found;
	## - swg_dsn_no_dataset_info_found: swg_dsn_error; if the
	##     dataset information collection can not be found or is empty.
	
	_local admin_view << swg_dsn_engine.get_admin_view()
# <= = = = = = = = = = = = = = = = = = = = possible condition
	.dataset_infos << admin_view.collections[:swg_dsn_dataset_info]
	_if quiet? ~= _true _andif
	    .dataset_infos _is _unset _orif
	    .dataset_infos.empty?
	_then
		condition.raise(:swg_dsn_no_dataset_info_found,:supplement,"")
# <============================================================
	_endif

	>> .dataset_infos

_endmethod

#------------------------------------------------------------------------------
# Method: check_as_built_path()   KGU
#------------------------------------------------------------------------------
_pragma(classify_level=restricted,
	topic={swg, swg_design,dsn_admin})
_pragma(classify_level=restricted, topic={swg, swg_design,dsn_admin})
_private _method swg_dsn_user_alternative_control.check_as_built_path(ds_info_record, _optional dataset)

	## D e s c r i p t i o n :
	## Checks, that in the given database view DATASET an alternative
	## with the name configured in the field 'as_built_alternative_path'
	## of DS_INFO_RECORD exists. (If not so, forces a condition.)
	## If DATASET is omitted, then the dataset with the name stored
	## in the DS_INFO_RECORD (field 'dataset_name') is checked.
	## 
	## A r g u m e n t s  (- <name>: <class>; <description>):
	## - DS_INFO_RECORD: swg_dsn_dataset_info; the info record corresponding
	##   to DATASET and expected to contain the as_built path name.
	## - DATASET: datbase_view; the view to be investigated;
	## 
	## D y n a m i c   V a r i a b l e s  (- <name>: <class>; <description>):
	## none
	## 
	## R e s u l t s  (- <name>: <class>; <description>):
	## none
	## 
	## S i d e   E f f e c t s   (incl. important slot changes):
	## A temporary replicate of DATASET is created and discarded afterwards.
	## 
	## C o n d i t i o n s  (- <name>: <type>; <indication>):
	## - db_thing_not_found: swg_dsn_error; if the wanted alternative
	##   is not found in the given DATASET or the dataset was not found.
	## 

	_if dataset _is _unset _andif (dataset << ds_info_record.referred_dataset) _is _unset
	_then
	# Attempt to get the dataset default failed.
		condition.raise(:db_thing_not_found, :type, :gis_ds_view, :name, ds_info_record.dataset_name.as_symbol())
	_elif (as_built_alt_path << ds_info_record.as_built_alternative_path) <> "|"
	_then
	# We only need to validate as-built path when it is not top as the top alternative always exists.
		repl << dataset.replicate_without_collections(0)
		_protect
		# If the as-built data alternative does not exist then condition db_thing_not_found will occur:
			repl.go_to_alternative(as_built_alt_path)
		_protection
			repl.discard()
		_endprotect
	_endif
_endmethod
$

#------------------------------------------------------------------------------
# Method: remove_reference_replicates()   KGU
#------------------------------------------------------------------------------
_pragma(classify_level=restricted,
	topic={swg, swg_design, dsn_admin, dsn_user_alt_ctrl})
_pragma(classify_level=restricted, topic={swg, swg_design, dsn_admin, dsn_user_alt_ctrl})
_private _method swg_dsn_user_alternative_control.remove_reference_replicates()
	##
	## D e s c r i p t i o n :
	## Examines all cached reference replicates and discards them.
	## 
	## A r g u m e n t s  (- <name>: <class>; <description>):
	## none
	## 
	## D y n a m i c   V a r i a b l e s  (- <name>: <class>; <description>):
	## none
	##
	## R e s u l t s  (- <name>: <class>; <description>):
	## none
	## 
	## S i d e   E f f e c t s   (incl. important slot changes):
	## Delivers occupied slots in ds_environment. Empties slot .replicates.
	## 
	## C o n d i t i o n s  (- <name>: <type>; <indication>):
	## none

	_lock _self
	      _protect
	      _protection # NOTE: The following code block has been made thread safe and non-interruptable
		      _for rpl_list _over .replicates.fast_elements()
		      _loop
			      _for rpl_vec _over rpl_list.fast_elements()
			      _loop
			      # for performance reasons it is not checked whether the view
			      # might already have been discarded by another agent. It is
			      # assumed that to discard twice does not make calamity. 
				      rpl_vec[1].discard()
			      _endloop
		      _endloop
		      
		      .replicates.empty()
	      _endprotect 
	_endlock 
_endmethod
$

#------------------------------------------------------------------------------
# Method: analyse_accessibility()   KGU
#------------------------------------------------------------------------------
_pragma(classify_level=restricted,
	topic={swg, swg_design, dsn_admin, dsn_user_alt_ctrl})
_private _method swg_dsn_user_alternative_control.analyse_accessibility(dataset_infos)
	## D e s c r i p t i o n :
	## Analyses for which of the given dataset infos there is an
	## accessible dataset and whether there are datasets with
	## unsaved changes or not.
	## A prerequisite is that the initial dataset checking via
	## _self.initialise() has already been done.
	## 
	## A r g u m e n t s  (- <name>: <class>; <description>):
	## - DATASET_INFOS: collection of swg_dsn_dataset_info; a set
	##     of dataset information records specifying the datasets
	##     something is to be done with.
	## 
	## D y n a m i c   V a r i a b l e s  (- <name>: <class>; <description>):
	## none
	## 
	## R e s u l t s  (- <name>: <class>; <description>):
	## - ACCESSIBILITY: property_list[symbol] of boolean; For each of
	##     the dataset_names referred to by the given DATASET_INFOS
	##     records, a boolean entry reports whether the requested
	##     dataset is available (_true) or not (_false);
	## - MISS_STRING: character_vector; A string containing an enumeration
	##     of all dataset names the dataset for which cannot be accessed;
	## - UNSAVED?: boolean; Is _true iff in one of the accessible
	##     datasets there are unsaved changes.
	## 
	## S i d e   E f f e c t s   (incl. important slot changes):
	## none
	## 
	## C o n d i t i o n s  (- <name>: <type>; <indication>):
	## - swg_dsn_design_system_not_working: swg_dsn_error; if the design
	##     system has not been started or was switched off.

	_local unsaved? << _false     # boolean; flag for unsaved changes
	_local miss_strm << internal_text_output_stream.new()
	                              # text_stream; collects missing dataset names.
	_local miss_string            # character_vector; text for conditions etc.
	
	_local ds_name                # symbol; name of a dataset
	_local ds_view                # gis_ds_view; an inspected dataset

	_local accessibility << property_list.new(dataset_infos.size)
	                              # property_list[symbol] of boolean; result

	_if _not .design_system_working?
	_then
		_self.design_system_working? << _true 
#		condition.raise(:swg_dsn_design_system_not_working)
	_endif
	
	_for ds_info _over dataset_infos.fast_elements()
	_loop
		ds_name << ds_info.dataset_name.as_symbol()
		accessibility[ds_name] << _true
		
		_if (ds_view << gis_program_manager.cached_dataset(ds_name)) _is _unset _orif
		    .view_status[ds_name] _is _unset
		_then
			# Note that creation won't be possible
			accessibility[ds_name] << _false
			# Fill in the condition parameter
			miss_strm.write(", ", ds_name)
		_elif ds_view.updated?
		_then
			# Set the flag
			unsaved? << _true
		_endif
		
	_endloop

	miss_string << miss_strm.string
	_if miss_string.size > 0
	_then
		miss_string << miss_string.subseq(3)
	_endif

	>> accessibility, miss_string, unsaved?

#------------------------------------------------------------------------------
# Method: go_as_deep_as_possible()   KGU
#------------------------------------------------------------------------------
_pragma(classify_level=restricted,
	topic={swg, swg_design, dsn_user_alt_ctrl})
_private _method swg_dsn_user_alternative_control.go_as_deep_as_possible(dataset_info, scheme_path, _optional a_view)
	## D e s c r i p t i o n :
	## Embeds the given SCHEME_PATH in the alternative tree of
	## the dataset described by the DATASET_INFO. If A_VIEW is
	## given then the alternative change will be performed in
	## A_VIEW instead. Of course, A_VIEW must be either the disk
	## version of the specified dataset or any replicate with or
	## without collections.)
	## The deepest alternative mappable to a scheme node within the
	## path is detected, and two ropes are returned:
	## ARRIVED_PATH is the alternative path to the detected spawning
	## alternative, REMAINING_PATH is the remaining part of the scheme
	## path the mapping of which to existing alternatives failed.
	## R e q u i r e m e n t s :
	## The view described by DATASET_INFO must be accessible and
	## correctly configured. A_VIEW, if given, must correspond to
	## the information in DATASET_INFO.
	## 
	## A r g u m e n t s  (- <name>: <class>; <description>):
	## - DATASET_INFO: swg_dsn_dataset_info (record); the description
	##     record from the design administration dataset for the
	##     dataset the path is to be searched in;
	## - SCHEME_PATH: list of sys_id; the sequence of scheme identifiers
	##     characterising the path from the base schem (root) to the
	##     interesting scheme;
	##     a scheme path with first element 0, however, designates
	##     a build-plan pool alternative;
	## - A_VIEW: database_view; the disk version of the dataset
	##     described by DATASET_INFO (if given then it must
	##     correspond to DATASET_INFO!).
	## 
	## D y n a m i c   V a r i a b l e s  (- <name>: <class>; <description>):
	## none
	## 
	## R e s u l t s  (- <name>: <class>; <description>):
	## - MAPPED_PATH: (list of (_unset/string))/_unset; the maximum
	##     alternative path that can be mapped with a prefix part of
	##     the SCHEME_PATH; if by some reason no mapping at all is
	##     possible then it will be _unset. Note that the path is an
	##     absolute path suitable for ds_version_view.go_to_alternative()
	##     i.e. its first element is _unset;
	## - REMAINING_PATH: list of sys_id; that suffix part of the
	##     scheme path that could not be mapped onto existing alternatives.
	## 
	## S i d e   E f f e c t s   (incl. important slot changes):
	## Temporarily a replicate view without collections is used.
	## 
	## C o n d i t i o n s  (- <name>: <type>; <indication>):
	## none

	_local dataset_name << dataset_info.dataset_name.as_symbol()
	_local mapped_path << _self.alt_path_from_scheme_path(dataset_info,scheme_path)
	_local arrived_path, remaining_path  # rope of (_unset/string); absolute alternative paths 
	
	_local arrived_level                 # integer

	#
	# care for correct setting of the A_VIEW.
	#
	_if a_view _is _unset
	_then
		a_view << gis_program_manager.cached_dataset(dataset_name)
	_endif
	
	mapped_path.add_first(_unset)
	
	# First try a fast optimistic test
	_if mapped_path.size < 2 _orif
	    _self.go_to_path_if_possible(a_view,mapped_path,:readonly)
	_then
		# The fast attempt succeeded!
		arrived_path << mapped_path
		remaining_path << {}
	_else		
		# The fast attempt failed, hence we may have to go down step by step
		# First check where we are (to avoid omittable version
		# changes). There is a considerable probability that we are
		# already at the spawning point.
		arrived_path << rope.new_with(_unset)
		arrived_path.add_all_last(a_view.alternative_path)

		arrived_level << arrived_path.size
		_for ix _over 2.upto(arrived_level.min(mapped_path.size))
		_loop
			_if arrived_path[ix] ~= mapped_path[ix]
			_then
				arrived_level << ix - 1
				_leave 
		# <===================
			_endif
		_endloop
		_if arrived_level < arrived_path.size
		_then
			a_view.go_to_alternative(arrived_level - arrived_path.size)
		_endif
		arrived_path << mapped_path.subseq(1,arrived_level)

		_if arrived_level < mapped_path.size
		_then
			_for alt_name _over mapped_path.subseq(arrived_level+1).fast_elements()
			_loop
				_if a_view.has_alternative?(alt_name.write_string)
				_then
					a_view.down(alt_name)
					arrived_path.add_last(alt_name)
				_else
					# there is no matching alternative: just break
					_leave
			# <===================
				_endif
			_endloop # for alt_name ...
			remaining_path << mapped_path.subseq(arrived_path.size + 1)
		_else
			remaining_path << {}
		_endif
	_endif 

	>> arrived_path, remaining_path
	
_endmethod
$

#------------------------------------------------------------------------------
# Method: go_to_path_if_possible()   KGU
#------------------------------------------------------------------------------
_pragma(classify_level=restricted,
	topic={swg, swg_design, dsn_user_alt_ctrl})
_pragma(classify_level=restricted, topic={swg, swg_design, dsn_user_alt_ctrl})
_private _method swg_dsn_user_alternative_control.go_to_path_if_possible(a_view, abs_alt_path, _optional mode)
	## D e s c r i p t i o n :
	## Tests the existence of the given absolute alternative path
	## ABS_ALT_PATH (i.e. it is expected to start with an _unset
	## element) in ds_version_view A_VIEW, and if it existes then
	## goes there, i.e. if the result is _true, then the version
	## will have changed, otherwise not.
	## 
	## A r g u m e n t s  (- <name>: <class>; <description>):
	## + A_VIEW: ds_version_view; an arbitrary version-managed view
	##     (may be a replicate);
	## - ABS_ALT_PATH: rope of {_unset/string}
	## - MODE: symbol; may be :hard_readonly, :readonly or :write.
        ##     If MODE is unset, then the access mode in the alternative
        ##     will be the mode which was current the last time A_VIEW was
        ##     at the alternative (if it had been there before), or else
        ##     A_VIEW's current mode.  (In all cases, if the mode would have
        ##     been writable, but this is not possible because another
        ##     user/view is already accessing the alternative for writing,
        ##     then A_VIEW's mode will be readonly.) If _self had been at
        ##     the alternative previously, and the selected mode is
        ##     readonly, and A_VIEW had been at a selected checkpoint (by
        ##     using go_to_checkpoint() with MODE = :readonly), then the
        ##     version of the alternative is the version at that
        ##     checkpoint.
	## 
	## D y n a m i c   V a r i a b l e s  (- <name>: <class>; <description>):
	## none
	## 
	## R e s u l t s  (- <name>: <class>; <description>):
	## - GONE_TO?: boolean; is _true if the path exists in the
	##     given view and is the active alternative path now,
	##     _false otherwise;
	## - CHANGED?: boolean; indicates whether the new version
	##     differs from the former one. Only returned if GONE_TO?
	##     is _true;
	## - WRITABLE?: boolean; indicates whether the new version
	##     is writable. Only returned if GONE_TO? is _true;
	## 
	## S i d e   E f f e c t s   (incl. important slot changes):
        ## A_VIEW is likely to chnage version, and if so, it sends
        ## change protocols :checkpoint_list, :alternative_list,
        ## and :alternative_path.
	## 
	## C o n d i t i o n s  (- <name>: <type>; <indication>):
	## none

	gone_to? << _false 
	_try 
		_if _not a_view.alternative_path.eq?(abs_alt_path)
		_then 
			(changed?, writable?) << _allresults a_view.go_to_alternative(abs_alt_path,mode)
			gone_to? << _true
		_elif _not a_view.mode _is mode
		_then 
			a_view.switch(mode)
		_endif 
		a_view.rollforward()
	_when db_thing_not_found
	_endtry
	
	>> gone_to?, changed?, writable?
_endmethod
$

# Method: create_remaining_path()   KGU
#------------------------------------------------------------------------------
_pragma(classify_level=restricted,
	topic={swg, swg_design, dsn_user_alt_ctrl})
_pragma(classify_level=restricted, topic={swg, swg_design, dsn_user_alt_ctrl})
_private _method swg_dsn_user_alternative_control.create_remaining_path(a_view, add_alt_path, _gather keys_and_args)

	## D e s c r i p t i o n :
	## Given a database view A_VIEW and a relative alternative path
	## ADD_ALT_PATH of for labelling alternatives, a new alternative branch
	## is created, starting beneath the current alternative with
	## one alternative per ADD_ALT_PATH element.
	## 
	## A r g u m e n t s  (- <name>: <class>; <description>):
	## - A_VIEW: database_view; the view the new alternative path
	##     is to be created in;
	## - ADD_ALT_PATH: list of string; the strings are used for the
	##     labelling down of the alternatives in the given sequence. 
	## - KEYS_AND_ARGS: alternating key/value pairs; Optional list
	##     of creation parameters, to be passed to the dataset
	##     alternative creation method.
	##     Refer to method CREATE_MAPPED_ALTERNATIVE() for details
	##     of valid options.
	## 
	## D y n a m i c   V a r i a b l e s  (- <name>: <class>; <description>):
	## none
	## 
	## R e s u l t s  (- <name>: <class>; <description>):
	## - COMPLETE_SUCCESS?: boolean; if the complete path was
	##     successfully created
	## 
	## S i d e   E f f e c t s   (incl. important slot changes):
	## none
	## 
	## C o n d i t i o n s  (- <name>: <type>; <indication>):
	## none
	_local complete_success? << _false
	_local creation_error_catch << "AltCreationError"  # a catch tag
	
	_catch creation_error_catch
	       _handling error _with
	       _proc(cond)
		       _import creation_error_catch
		       # Inform about the error and deny creation success. 
		       condition.raise(:information,
				       :string, cond.report_contents_string)
		       _throw creation_error_catch
	       _endproc
	       
	       _for alt_name _over add_alt_path.fast_elements()
	       _loop
		       _self.create_mapped_alternative(a_view, alt_name, _scatter keys_and_args)
	       _endloop
	       # Note that the creation has completed
	       complete_success? << _true 
	_endcatch # creation_error_catch
	
	>> complete_success?
_endmethod

#------------------------------------------------------------------------------
# Method: create_mapped_alternative()   KGU
#------------------------------------------------------------------------------
_pragma(classify_level=restricted,
	topic={swg, swg_design, dsn_user_alt_ctrl})
_pragma(classify_level=restricted, topic={swg, swg_design, dsn_user_alt_ctrl})
_private _method swg_dsn_user_alternative_control.create_mapped_alternative(a_view, alt_name, _gather keys_and_args)

	## D e s c r i p t i o n :
	## Spawns the current alternative of dataset A_VIEW, giving the
	## new alternative the name ALT_NAME as derived from the
	## scheme's id according to the mapping function id_to_alt_name().
	## N o t e :
	## In contrast to ds_version_view.spawn(), it   d o e s   descend
	## to the created alternative!
	## If there has already been an alternative with name ALT_NAME, then
	## a new alternative won't be created (but the view goes down).
	## The meaning of the arguments SAME_FILE?, WHERE_TO_CREATE, and
	## FILE_NAMES is equal to those of database_view.spawn().
	## 
	## A r g u m e n t s  (- <name>: <class>; <description>):
	## - A_VIEW: database_view; The database view where the alternative is
	##     to be created;
	## - ALT_NAME: string; the proposed alternative name;
	## - KEYS_AND_ARGS: alternating key/value pairs; Optional list
	##     of creation parameters, to be passed to the dataset
	##     alternative creation method. Valid options (also refer to
	##     ds_version_view.spawn()) are as follows:
	##
	##   General parameters:
	##
	##      :same_file?, boolean
	##         decides whether to create the new alternative in the
	##         same file as the current alternative (:same_file? =_TRUE)
	##         or in a file determined by :where_to_create and
	##         :file_names (:same_file? = _FALSE).
	##         Default is _TRUE.
	##      :where_to_create, integer/byte_character_vector
	##         if an integer, it is interpreted as creation_index
	##         into _self's searchpath, otherwise it must be the
	##         directory pathname in which to put the new file(s);
	##         case same_file? = _TRUE it is ignored, otherwise
	##         mandatory!
	##      :file_names, vector of byte_character_vector
	##         the names of the files to put the new alternative
	##         in, corresponding to the view's current set of files.
	##
	##    Design Clerk parameters:
	##
	##      :parent_checkpoint, string
	##         The name of a checkpoint in the parent to act as base for
	##         the new alternative or design. If not supplied, the disk 
	##         version in the parent alternative is used.
	##    
	##      :design_impl, symbol
	##         Defines how the design is to be implemented - this may be
	##         one of
	##            :real
	##               A real alternative (as opposed to a design) will be
	##               created, even if the parent has been registered as a
	##               design parent. 
	##            :working_alternative
	##               A design will be created, whether or not the parent 
	##               has been registered as a design parent. If the design 
	##               is not accessed for a protracted period and design_clerk 
	##               gc'ing is run, the design will be 'archived', saving 
	##               space, to be restored automatically when it is gone to
	##               next time.
	##            :permanent
	##               Like :working_alternative, except the design will
	##               not be archived by default, even if it has been
	##               on-line for a protracted period. It may be marked
	##               as ready for archiving by using
	##               .set_design_status(). 
	##            :scratch
	##               Creates a design which is managed in a scratch
	##               alternative, reducing version management overheads.
	##               Note that this option is not recommended without
	##               thorough understanding of how scratch designs should
	##               be controlled and managed.
	##         If this option is not supplied, it defaults to
	##         :working_alternative if the current alternative is 
	##         registered as a design parent and :real if it isn't.  
	##         If the current alternative is a design, then the new
	##         alternative is always a design. 
	## 
	## D y n a m i c   V a r i a b l e s  (- <name>: <class>; <description>):
	## none
	## 
	## R e s u l t s  (- <name>: <class>; <description>):
	## - ALT_NAME: string; the name of the created new alternative.
	## 
	## S i d e   E f f e c t s   (incl. important slot changes):
	## If a new alternative is created, then in the current alternative (the
	## parent of the new one) a checkpoint is set with name "{<alt_name>}"
	## where <alt_name> is the name of the new alternative.
        ## Change protocol :alternative_list is sent by A_VIEW if the alternative
        ## is successfully created.
	## 
	## C o n d i t i o n s  (- <name>: <type>; <indication>):
	## Those of database_view.spawn(): auth!alternative_locked.

	_if _not a_view.has_alternative?(alt_name.write_string)
	_then
		dc_args << property_list.new_with(_scatter keys_and_args)

		same_file?      << dc_args.remove_key(:same_file?).default(_true)
		where_to_create << dc_args.remove_key(:where_to_create)
		file_names      << dc_args.remove_key(:file_names)

		a_view.rollforward()
		a_view.spawn(alt_name, same_file?, where_to_create, file_names, _scatter dc_args)
	_endif
	
	a_view.down(alt_name)
	
	>> alt_name
_endmethod
$

#------------------------------------------------------------------------------
# Method: id_to_alt_name()   KGU
#------------------------------------------------------------------------------

_pragma(classify_level=restricted, topic={swg, swg_design, dsn_user_alt_ctrl})
_method swg_dsn_user_alternative_control.id_to_alt_name(scheme_id, dataset_name)
	## D e s c r i p t i o n :
	## Maps the given scheme key SCHEME_ID (a unique value
	## allocator) to an appropriate alternative name on the
	## database view A_VIEW.
	## 
	## A r g u m e n t s  (- <name>: <class>; <description>):
	## - SCHEME_ID: unsigned integer (sys_id): The unique numeric
	##     key of the planning scheme; SCHEME_ID must not be 0;
	## - DATASET_NAME: the name of the dataset in which the
	##     alternative resides.
	## 
	## D y n a m i c   V a r i a b l e s
	##
	## The following two dynamics,!RELOCATING_SCHEME_ID! and
	## !RELOCATING_TEMP_SCHEME_ID!, are for resolving the
	## conflict of mapping two schemes, an original scheme whose 
	## alternatives are being relocated to a new, non-pool managed,
	## location in the design alternative hierarchy and the
	## temporary scheme created and associated with the new
	## alternatives whilst a far-merge process between the original
	## scheme and the temporary scheme is used to transfer the
	## scheme changes to their new location. On completion of the
	## transfer, the temporary scheme is deleted and the original
	## scheme refers to the new alternatives by the default
	## behaviour that the original scheme id is encoded into the
	## alternative name. These dynamics are therefore set, or not,
	## to indicate whether SCHEME_ID is to map to the alternative
	## name based on !RELOCATING_SCHEME_ID! or on SCHEME_ID.
	##
	## Until the the relocation has successfully completed, both of
	## these dynamics must be set whenever this method is called so
	## that the returned ALTERNATIVE_NAME will be the name:
	##     "_SELF.ALT_NAME_PREFIX" + "_!RELOCATING_SCHEME_ID!" 
	## rather than the name default derived from SCHEME_ID whcih
	## will be the id of the temporary scheme. Once the relocation
	## has completed SCHEME_ID will be the id of the the relocated
	## scheme (ie. !RELOCATING_SCHEME_ID!)
	##
	## !RELOCATING_SCHEME_ID! either _UNSET or <a scheme_id>
	##
	##     If set, then SCHEME_PATH is the path of a temporary
	##     scheme created in order to relocate the scheme with
	##     A_SCHEME_ID (as specified by this dynamic) by a far-merge
	##     operation into a new, non-pool managed, alternative
	##     structure being created for SCHEME_PATH such that the
	##     scheme alternatives in the new location will be named
	##     with their alternative name derived from A_SCHEME_ID,
	##     rather than SCHEM_PATH.LAST, such that the original
	##     scheme can be linked to the new alternative path when
	##     the temporary scheme is deleted on completion of the
	##     far-merge operation. During the relocation, this dynamic
	##     must be set so that the alternative path for the
	##     temporary scheme is derived using the id of the
	##     scheme being relocated.
	##
	## !RELOCATING_TEMP_SCHEME_ID! either _UNSET or <a scheme_id>
	##
	##     If set, then ALT_NAME must be the name of a non-pool
	##     managed alternative currently belonging to a temporary
	##     scheme undergoing construction (as a result of another
	##     scheme being relocated by a far merge of its changes to
	##     ALT_NAME). Whilst this is taking pace, as indicated by
	##     this and the above dynamic being set, this alternative
	##     must map back to the temporary scheme. On completion of
	##     the far merge the temporary scheme will be deleted and this
	##     alternative will map back to the relocated scheme by the
	##     normal mechanism that the alternative name includes its
	##     scheme id.
	##
	##     The effect of setting this (and the above) dynamic is
	##     therefore to map SCHEME_ID to the alternative name based
	##     on the id (!RELOCATING_SCHEME_ID!) of the scheme
	##     currently being relocated rather than the id SCHEME_ID
	##     (the default) which is !relocating_temp_scheme_id!,
	##     the id of the temporary scheme/
	## 
	## R e s u l t s  (- <name>: <class>; <description>):
	## - ALTERNATIVE_NAME: string; the mapped alternative name.
	## 
	## S i d e   E f f e c t s   (incl. important slot changes):
	## none
	## 
	## C o n d i t i o n s  (- <name>: <type>; <indication>):
	## none

	# If the specified design is a top-level design then we see if
	# there is a pool managed alternative assigned to it and return
	# the name of the alernative created for the design by the pool
	# manager.
	#
	# If the specified scheme id is !RELOCATING_TEMP_SCHEME_ID!
	# and !RELOCATING_SCHEME_ID! is also specified then return
	# an alternative name that is the concatenation of the
	# alternative name prefix _SELF.ALT_NAME_PREFIX with the write
	# string of the id !RELOCATING_SCHEME_ID!
	#
	# Otherwise The name is the concatenation of
	# _SELF.ALT_NAME_PREFIX with the write string of the argument
	# SCHEME_ID.

	_dynamic !relocating_scheme_id!      # If set, the ID of the scheme being relocated via the temporary scheme
		                                # with ID !relocating_temp_scheme_id! (see method comment above)
	_dynamic !relocating_temp_scheme_id! # If set, the ID of the temporary scheme, occupying alternative ALT_NAME,
		                                # created for the operation of relocation the scheme with id
		                                # specified by !relocating_scheme_id! (see method comment above)

	>> _if (alt_name << swg_dsn_alt_pool_manager.alternative_name_for_design(scheme_id, dataset_name)) _isnt _unset
	   _then
	   # Pool managed top-level design
		   >> alt_name
	   _elif !relocating_scheme_id! _isnt _unset _andif scheme_id _is !relocating_temp_scheme_id!
	   _then 
	   # Alternative map for a temporary scheme created whilst
           # relocating the scheme with id sw$relocating_scheme_id!
		   >> write_string(_self.alt_name_prefix, !relocating_scheme_id!)
	   _else
		   >> write_string(_self.alt_name_prefix, scheme_id)
	   _endif 
_endmethod
$

_pragma(classify_level=restricted, topic={swg, swg_design, dsn_user_alt_ctrl})
_method swg_dsn_user_alternative_control.alt_name_to_id(a_view, _optional alt_name)
	## D e s c r i p t i o n :
	## Checks to see if the specified alternative is a pool managed
	## alternative for a design and if so then returns the scheme
	## id for the design assigned to it, otherwise extracts the
	## encrypted scheme id out of the alternative name ALT_NAME (by
	## default the name of the current alternative) of the database
	## view A_VIEW.
	## 
	## A r g u m e n t s  (- <name>: <class>; <description>):
	## - A_VIEW: ds_version_view: The view, the alternative name of
	##     which is to be converted (this is for the case an
	##     adaptive mapping is relaised);
	## - ALT_NAME: string; the alternative name to be converted
	##     into a planning scheme key. Default is the name of the
	##     current alternative of A_VIEW (last component of path).
	## 
	## D y n a m i c   V a r i a b l e s:
	##
	## The following two dynamics,!RELOCATING_SCHEME_ID! and
	## !RELOCATING_TEMP_SCHEME_ID!, are for resolving the
	## conflict of mapping two schemes, an original scheme whose 
	## alternatives are being relocated to a new, non-pool managed,
	## location in the design alternative hierarchy and the
	## temporary scheme created and associated with the new
	## alternatives whilst a far-merge process between the original
	## scheme and the temoprary scheme is used to transfer the
	## scheme changes to their new location. On completion of the
	## transfer, the temporary scheme is deleted and the original
	## scheme refers to the new alternatives by the default
	## behaviour that the original scheme id is encoded into the
	## alternative name. These dynamics are therefore set, or not,
	## to indicate whether ALT_NAME is to map to the temporary
	## scheme or (on completion of the relocate) the original
	## scheme being relocated.
	##
	## Until the the relocation has successfully completed, both of
	## these dynamics must be set whenever this method is called so
	## that ALT_NAME will map to the temporary scheme rather than
	## the scheme that is still undergoing relocation.
	##
	## !RELOCATING_SCHEME_ID! either _UNSET or <a scheme_id>
	##
	##     If set, then ALT_NAME is name of an alternative still
	##     associated with a temporary scheme, created in order to
	##     relocate the scheme with ID !RELOCATING_SCHEME_ID!,
	##     by a far-merge operation into the non-pool managed
	##     alternative, named ALT_NAME, that has been named
	##     "_SELF.ALT_NAME_PREFIX" + "_!RELOCATING_SCHEME_ID!" 
	##     such that the original scheme being relocated can be
	##     linked to this alternative when the temporary scheme is
	##     deleted on completion of the far-merge operation.
	##     ALT_NAME must therefore embed the id specified by
	##     !RELOCATING_SCHEME_ID!
	##
	##     Also refer to the following related dynamic.
	##
	## !RELOCATING_TEMP_SCHEME_ID! either _UNSET or <a scheme_id>
	##
	##     If set, then ALT_NAME must be the name of a non-pool
	##     managed alternative currently belonging to a temporary
	##     scheme undergoing construction (as a result of another
	##     scheme being relocated by a far merge of its changes to
	##     ALT_NAME). Whilst this is taking pace, as indicated by
	##     this and the above dynamic being set, this alternative
	##     must map back to the temporary scheme. On completion of
	##     the far merge the temporary scheme will be deleted and this
	##     alternative will map back to the relocated scheme by the
	##     normal mechanism that the alternative name includes its
	##     scheme id.
	##
	##     The effect of setting this (and the above) dynamic is
	##     therefore to map ALT_NAME to the temporary scheme,
	##     rather than the scheme being relocated (the default), by
	##     simply returning the value of this shared variable.
	## 
	## R e s u l t s  (- <name>: <class>; <description>):
	## - SCHEME_ID: unsigned integer / _unset; the extracted scheme
	##     identifier if the given ALT_NAME may be a planning
	##     alternative name. Otherwise the result will be _unset.
	## 
	## S i d e   E f f e c t s   (incl. important slot changes):
	## none
	## 
	## C o n d i t i o n s  (- <name>: <type>; <indication>):
	## none

	# Ensure that ALT_NAME is a string. If we are at the top level,
	# an empty string is set, because the top alternative may never
	# be a scheme alternative.
	#?? What about an extract / replicate?

	_dynamic !relocating_scheme_id!      # If set, the ID of the scheme being relocated via the temporary scheme
		                                # with ID !relocating_temp_scheme_id! (see method comment above)
	_dynamic !relocating_temp_scheme_id! # If set, the ID of the temporary scheme, occupying alternative ALT_NAME,
		                                # created for the operation of relocation the scheme with id
		                                # specified by !relocating_scheme_id! (see method comment above)
	

	_local alt_name_prefix << _self.alt_name_prefix     # string; the standard prefix
	# for alternative names for schemes
	_local prefix_length << alt_name_prefix.size
		     
	_if alt_name _is _unset
	_then
		_if a_view.alternative_level > 0
		_then
			alt_name << a_view.alternative_path.last
		_else
			alt_name << ""
		_endif
	_endif
	
	_if (id << swg_dsn_alt_pool_manager.design_for_alternative(a_view.dataset_name, alt_name)) _isnt _unset
	_then
		# Pool managed design alternative
		scheme_id << _if id <> 0
			     _then >> id
			     _else >> _unset # free pool alternative
			     _endif 
	_elif alt_name.size >= prefix_length _andif
	      alt_name.subseq(1,prefix_length) = _self.alt_name_prefix
	_then
	# Non-pool managed alternative
		_if !relocating_scheme_id! _isnt _unset _andif
		    (scheme_id << alt_name.subseq(prefix_length+1).as_number(_true)) = !relocating_scheme_id!
		_then
		# A relocate of a scheme into ALT_NAME is currently taking
		# place so this alternative must map to the temporary scheme,
		# as specified by !RELOCATING_TEMP_SCHEME_ID!, created for
		# the relocation operations.
			scheme_id << !relocating_temp_scheme_id!
		_else 
			scheme_id << alt_name.subseq(prefix_length+1).as_number(_true)
		_endif 
	_endif

	_if scheme_id _isnt _unset _andif (scheme_id < 0 _orif _not {:integer,:bignum}.includes?(scheme_id.class_name))
	_then
		scheme_id << _unset
	_endif

	>> scheme_id
	
_endmethod

# Method: decompose_alternative_name()  KGU
#------------------------------------------------------------------------------
_pragma(classify_level=restricted,
	topic={swg, swg_design, dsn_user_alt_ctrl})
_method swg_dsn_user_alternative_control.decompose_alternative_name(name)
	## D e s c r i p t i o n :
	## Return a vector containing the path components of NAME. If
	## NAME starts with '|', then the first element will be _unset,
	## to specify that the name is absolute. If no name is
	## specified, returns an empty vector. If a malformed name was
	## given, returns _unset.
	##
	## A r g u m e n t s  (- <name>: <class>; <description>):
	## - NAME: string; Alternative name to be decomposed
	## 
	## D y n a m i c   V a r i a b l e s  (- <name>: <class>; <description>):
	## none
	## 
	## R e s u l t s  (- <name>: <class>; <description>):
	## - RESULT: vector/_unset; vector containing the path
	##   components of NAME, empty vector if no name is specified,
	##   _unset in case a malformed name was  given 
	## 
	## S i d e   E f f e c t s   (incl. important slot changes):
	## none
	##
	## C o n d i t i o n s  (- <name>: <type>; <indication>):
	## none 

	
	# Remark: This method is "spied out" from ds_version_view
	_local alt_separator << ds_version_view.alt_separator # character
	_local end << name.size     # integer
	_local result << _unset     # (vector of string)/_unset; see above
	_local prev << 1            # integer; position of previous element
	_local index                # integer

	_if end > 0
	_then
		result << rope.new()
		name << name.as_charvec()   # ensure that we won't get a ds_byte_character_vector
		_if name[1] _is alt_separator
		_then
			prev << 2
			result.add(_unset)
			
			_if end _is 1
			_then
				_return result
# <===========================================
			_endif 
		_endif
		
		_loop
			index << name.index_of(alt_separator, prev).default(end+1)
			
			_if prev <= index-1
			_then
				result.add(name.slice(prev, index-1))
			_else
				# Malformed alternative path name
				_return _unset
# <===========================================
			_endif 
			
			_if index = end+1
			_then _leave
		# <=================
			_endif 
			
			prev << index+1
		_endloop 
	_endif 
	>> result
_endmethod

#------------------------------------------------------------------------------
# Method: do_request_replicate()   KGU
#------------------------------------------------------------------------------
_pragma(classify_level=restricted,
	topic={swg, swg_design, dsn_user_alt_ctrl})
# KGU 19991012: argument FOR_RENDERING? added, algorithm redesigned.
_pragma(classify_level=restricted, topic={swg, swg_design, dsn_user_alt_ctrl})
_private _method swg_dsn_user_alternative_control.do_request_replicate(a_view, _optional for_rendering?)
	##
	## D e s c r i p t i o n :
	## Requests a (read-only) replicate view for the given dataset
	## that refers to a reference version of the scheme at the
	## current alternative.
	## 
	## A r g u m e n t s  (- <name>: <class> # <description>):
	## - A_VIEW: database_view # the source view;
	## - FOR_RENDERING?: boolean # if given, controls the exclusiveness
	##     of usage: _true means that the replicate is requested
	##     for rendering and a subsequent rendering request may
	##     reuse it without intermediate release, _false (the default)
	##     results in an exclusive reservation (only after an explicit
	##     release the replicate may be recycled).
	## 
	## D y n a m i c   V a r i a b l e s  (- <name>: <class>; <description>):
	## none
	## 
	## R e s u l t s  (- <name>: <class>; <description>):
	## - REPLICATE: database_view; The replicate view of A_VIEW
	##     The returned view will always be in read-only mode.
	## 
	## S i d e   E f f e c t s   (incl. important slot changes):
	## .replicates will be updated appropriately---maybe a released
	## replicate will be recycled.
	## ds_environment may be charged with further pcls/slots (this
	## might eventually lead to a database shutdown!).
	## 
	## C o n d i t i o n s  (- <name>: <type>; <indication>):
	## ds_environment overflow --> database shutdown is possible.

	_local dataset_name << a_view.dataset_name
	_local used_replicates     # list of {database_view,kleenean}; replicate cache list.
	_local replicate           # database_view; the result
	_local repl_entry          # {database_view, kleenean}; a replicate of a_view and release flag

	for_rendering? << for_rendering?.default(_false)
	
	_if dataset_name _is _unset
	_then
	# It is a replicate: we look for the view with same original dataset name
		dataset_name << _self.identify_dataset_of_replicate(a_view)
	_endif

	# Now look, whether perhaps the given view is a free replicate
	repl_entry << _self.find_replicate_entry(a_view,dataset_name)

	# If there has never been a replicate for this dataset then
	# create a new list in the table.

	_lock _self 
	      _protect
	      _protection # NOTE: The following code block has been made thread safe and non-interruptable
		      _if (used_replicates << .replicates[dataset_name]) _is _unset
		      _then
			      .replicates[dataset_name] << used_replicates << rope.new()
		      _endif
		      
		      # If the given view (A_VIEW) is not a cached replicate or it is
		      # not free then look for another re-usable (i.e. released or idle)
		      # replicate.
		      _if repl_entry _is _unset _orif
			  _not (repl_entry[2] _is _true _orif
				for_rendering? _andif repl_entry[2] _is _maybe)
		      _then
			      # Ensure unambiguous start situation
			      repl_entry << _unset
			      
			      # If this request is for rendering then a pending rendering
			      # replicate (with 2nd vector element being _maybe) must have
			      # priority. If there is one then repl_entry will point to its
			      # cache entry afterwards.
			      _for r_entry _over used_replicates.fast_elements()
			      _loop
				      _if r_entry[2] _is _true
				      _then
					      repl_entry << r_entry
					      
					      _if _not for_rendering?
					      _then
						      # We needn't go on looking for a possible
						      # pending entry which would have priority 
						      _leave
						      # <===========================
					      _endif
				      _elif for_rendering? _andif r_entry[2] _is _maybe
				      _then
					      repl_entry << r_entry
					      _leave
					      # <===================
				      _endif
			      _endloop
			      
		      _endif
		      
		      # If there is no re-usable entry then create a new replicate
		      # and cache it, otherwise just prepare the existing one. 
		      _if repl_entry _is _unset
		      _then
			      replicate << a_view.replicate(0,_true)
			      repl_entry << {replicate,_false}  # 2nd element will be refined below
			      used_replicates.add(repl_entry)
		      _else	
		      # It is, so return itself and mark it as occupied
			      replicate << repl_entry[1]
		      _endif
		      
		      # Refinement of the 2nd entry element (meaning <free??>)
		      repl_entry[2] << _if for_rendering?
				       _then >> _maybe
				       _else >> _false
				       _endif
		      
		      replicate.auth_view << _unset 
	      _endprotect 
	_endlock 
	
	>> replicate
	
_endmethod
$

# Method: do_move_replicate()   KGU
#------------------------------------------------------------------------------
_pragma(classify_level=restricted, topic={swg, swg_design, dsn_user_alt_ctrl})
_private _method swg_dsn_user_alternative_control.do_move_replicate(a_replicate, alt_path, reference, _optional raise_error_if_not_found?)
	## D e s c r i p t i o n :
	## Moves the given replicate view A_REPLICATE to the reference
	## version selected by REFERENCE of the scheme at the given
	## alternative path ALT_PATH.
	##
	## A r g u m e n t s  (- <name>: <class>; <description>):
	## + A_REPLICATE: ds_version_view; the replicate of A_VIEW that
	##     is to be moved. Note, that A_REPLICATE will be modified!
	## - ALT_PATH: _unset/rope of (_unset/string), an absolute alternative
	##     path where it doesn't matter whether the first element is
	##     _unset or not; if ALT_PATH is _unset as a whole then the
	##     current alternative path of A_REPLICATE is used. If the
	##     specified alternative is not found then A_REPLICATE is
	##     navigated as far down the specified path as could be
	##     reached and an error also raised according to the value
	##     of RAISE_ERROR_IF_NOT_FOUND?.
	##
	## - REFERENCE: symbol; a selector denoting the version the
	##     replicate view is forced to go to:
	##       :current means the current disk version and
	##       :base the starting version of the very scheme.
	## - RAISE_ERROR_IF_NOT_FOUND? If _TRUE (DEFAULT = _FALSE) then
	##       the error conditon :swg_dsn_err_missing_alternative will
	##       be raised if alternative specified by ALT_PATH is not
	##       found 
	## D y n a m i c   V a r i a b l e s  (- <name>: <class>; <description>):
	## none
	## 
	## R e s u l t s  (- <name>: <class>; <description>):
	## none
	## 
	## S i d e   E f f e c t s   (incl. important slot changes):
	## A_REPLICATE will be modified.
	## 
	## C o n d i t i o n s  (- <name>: <type>; <indication>):
	## none

	_local resulting_path

	_if alt_path _isnt _unset
	_then
		# copy the path lest we should confuse a dataset by changing
		# its alternative path rope.
		alt_path << rope.new_from(alt_path)
		# unify shape of path
		_if alt_path.empty? _orif
		    alt_path.first _isnt _unset
		_then
			alt_path.add_first(_unset)
		_endif 

		# Try to get to the target alternative with a single step
		_try 
		       a_replicate.go_to_alternative(alt_path)
		_when db_thing_not_found
			       # The direct jump failed, so go step by step towards the target
			       # alternative, as deep as possible.
			       _for alt_name _over alt_path.fast_elements()
			       _loop
				       _if alt_name _is _unset
				       _then
					       a_replicate.goto_top_alternative()
					       
				       _elif a_replicate.has_alternative?(alt_name.write_string)
				       _then
					       a_replicate.down(alt_name)
				       _else
					       _leave
				# <==================
				       _endif
				       
			       _endloop
			_if raise_error_if_not_found? _is _true 
			_then
			       ds_info << _self.dataset_info(a_replicate.original_dataset_name)
			       scheme_path << _self.scheme_path_from_alt_path(ds_info, alt_path)
			       job_type_name << _if (sch << _self.scheme(scheme_path)) _isnt _unset
						_then >> sch.project.job_type
						_else >> ""
						_endif
			       
			       condition.raise(:swg_dsn_err_missing_alternative, :dataset_name, a_replicate.external_name, :job_type_name, job_type_name)
			       
			_endif 
		_endtry 

		
		# If the wanted alternative does not exist i.e. the current
		# alternative of A_REPLICATE is above the target then there
		# will not be a base version of the wanted alternative, either.
		# So, go to the disk version in this case.
		_if (resulting_path << a_replicate.alternative_path).size < alt_path.size-1
		_then
			reference << :current
		_endif
	_else
		# We will need the last element of the path for the starting
		# checkpoint management in the following step
		alt_path << {a_replicate.alternative_path.last}
	_endif # if alt_path _isnt _unset

	_if reference _is :base
	_then
		#a_replicate.go_to_base_version()
		# KPM 2002/05/28 - CR 1769 - Replace line above with line below
		a_replicate.up_base()
	_else
		# the remaining case should be :current and for this only a
		# rollforward() is left to be done.
		a_replicate.rollforward()
	_endif

_endmethod
$

#------------------------------------------------------------------------------
# Method: do_release_replicate()
#------------------------------------------------------------------------------
_pragma(classify_level=restricted,
	topic={swg, swg_design, dsn_user_alt_ctrl})
_pragma(classify_level=restricted, topic={swg, swg_design, dsn_user_alt_ctrl})
_private _method swg_dsn_user_alternative_control.do_release_replicate(repl_view)
	##
	## D e s c r i p t i o n :
	## Cares for the recycling of the given replicate view
	## REPL_VIEW in the appropriate way, i.e. if the number of
	## administrated replicates for the same dataset is greater
	## than constant _self.max_repl_per_dataset, then REPL_VIEW
	## will be discarded, otherwise it is set to 'free'.
	## 
	## A r g u m e n t s  (- <name>: <class>; <description>):
	## + REPL_VIEW: database_view; The replicate no longer needed;
	## 
	## D y n a m i c   V a r i a b l e s  (- <name>: <class>; <description>):
	## none
	## 
	## R e s u l t s  (- <name>: <class>; <description>):
	## none
	## 
	## S i d e   E f f e c t s   (incl. important slot changes):
	## may change the number of used ds_environment slots
	## 
	## C o n d i t i o n s  (- <name>: <type>; <indication>):
	## none

	_local used_replicates     # list of {<replicate>, <free??>}
	
	# First find it in replicates, then decide whether to discard
	# it or
	# mark it as free
	# First identify the dataset and get the replicate list.

	dataset_name << _self.identify_dataset_of_replicate(repl_view)

	# If the replicate list is identified then decide whether to
	# discard or to mark free.

	_lock _self 
	      _protect
	      _protection # NOTE: The following code block has been made thread safe and non-interruptable
		      _if (used_replicates << .replicates[dataset_name]) _isnt _unset
		      _then
			      _for repl_ix, repl_entry _over used_replicates.fast_keys_and_elements()
			      _loop
				      _if repl_entry[1] _is repl_view
				      _then
					      _if used_replicates.size > _self.max_free_repl_per_dataset
					      _then
						      # If there is plenty of replicates then discard it.
						      repl_view.discard()
						      used_replicates.remove_nth(repl_ix)
					      _else
						      # Simply mark it as free if there are not more than
						      # the limit.
						      repl_entry[2] << _true    # free
						      
						      pred1 << predicate.eq(:dataset_name, dataset_name)
						      ds_info << .dataset_infos.select(pred1).an_element()
						      
						      _if ds_info _isnt _unset # shouldn't happen, but better safe than wrong
						      _then
							      altpath << ds_info.design_top_alternative_path()
							      repl_view.go_to_alternative(altpath, :readonly)
						      _else
							      # couldn't find the design-top, so just go to top.
							      repl_view.goto_top_alternative()
							      repl_view.switch(:readonly) # <-- just to be sure
						      _endif
						      
					      _endif
					      _leave
					      # <===================
				      _endif
			      _endloop
		      _endif
	      _endprotect
	_endlock 	      
_endmethod
$

#------------------------------------------------------------------------------
# Method: investigate_alternatives()
#------------------------------------------------------------------------------
_pragma(classify_level=restricted,
	topic={swg, swg_design, dsn_user_alt_ctrl})
_private _method swg_dsn_user_alternative_control.investigate_alternatives(a_view)
	## D e s c r i p t i o n :
	## Recursively investigates for all alternatives downwards, whether
	## there are current "occupants", i.e. current readers or writers. If
	## so then the respective information is gathered in the list OCCUPIED.
	## 
	## A r g u m e n t s  (- <name>: <class>; <description>):
	## - A_VIEW: ds_version_view; a view of the dataset to be investigated
	##     (typically a replicate without collections).
	## 
	## D y n a m i c   V a r i a b l e s  (- <name>: <class>; <description>):
	## 
	## R e s u l t s  (- <name>: <class>; <description>):
	## - OCCUPIED: list of {integer, string}; elements are
	##     pairs of scheme identifiers extracted from the
	##     alternative name and the occupant's login name for each
	##     alternative that has a current reader or writer;
	## 
	## S i d e   E f f e c t s   (incl. important slot changes):
	## Varying consumption of pcls.
	## 
	## C o n d i t i o n s  (- <name>: <type>; <indication>):
	## none

	_local occupied << rope.new()             # rope of {sys_id,string}; list of pairs of scheme id and login name
	_local occupant                           # string; a login name
	_local level << a_view.alternative_level  # remember the entering level
	_local alt_name                           # string, alternative_name
	
	_for alt _over a_view.alternatives.fast_elements()
	_loop
		alt_name << alt.alternative_name
		a_view.down(alt_name)
		_protect
			occupied.add_all_last(_self.investigate_alternatives(a_view))
		_protection
			# going up until the original alternative level is reached
			_over (a_view.alternative_level-1).downto(level)
			_loop
				a_view.up()  
			_endloop
		_endprotect 
	_endloop
	
	_if (occupant << a_view.current_writer) _isnt _unset _orif
	    a_view.current_readers.size > 1  # "yourself" i.e. will always be reader
	                                     # of the current alternative if not even
					     # the writer.
	_then
		_if occupant _is _unset
		_then
			# Find a reading occupant except "yourself"
			_for reader _over a_view.current_readers.fast_elements()
			_loop
				# at most one shot can be in vain
				_if (occupant << reader) ~= "yourself"
				_then
					_leave
			# <===================
				_endif
			_endloop
		_endif 
		occupied.add_last({_self.alt_name_to_id(a_view),
			      occupant})
	_endif
		
	>> occupied
_endmethod
$

#------------------------------------------------------------------------------
# Method: check_alternative_occupation_and_sub_tree()
#------------------------------------------------------------------------------
_pragma(classify_level=restricted,
	topic={swg, swg_design, dsn_user_alt_ctrl})
_pragma(classify_level=restricted, topic={swg, swg_design, dsn_user_alt_ctrl})
_private _method swg_dsn_user_alternative_control.check_alternative_occupation_and_sub_tree(seek_result, scheme_path, allow_sub_tree?)
	## D e s c r i p t i o n :
	## Checks the occupation of the alternative associated to SCHEME_PATH
	## by other readers or writers in all datasets under the name of which a
	## _true entry is found in SEEK_RESULT. If ALLOW_SUB_TREE? is _false
	## then existing sub-alternatives will cause an error condition,
	## otherwise the whole sub-tree is investigated as well.
	## Causes an error condition instead of returning when amn alternative
	## has a current reader or writer.
	## Note that most entries in SEEK_RESULTS will be modified, changing
	## structure to
	##    property_list[symbol] of (boolean / rope of (_unset/string)):
	## if the alternative to be removed is found, then the entry will be
	## replaced by the absolute alternative path.
	## 
	## A r g u m e n t s  (- <name>: <class>; <description>):
	## + SEEK_RESULT: property_list of boolean; the keys of all _true
	##     entries are the names of the found datasets to be inspected;
	## - SCHEME_PATH: list of integer; a sequence of swg_dsn_scheme
	##     keys, according to the scheme path down from root (top
	##     scheme) to the very scheme; 
	##     a scheme path with first element 0, however, designates
	##     a build-plan pool alternative;
	## - ALLOW_SUB_TREE?: boolean; if _false, then a condition will
	##     be raised as soon as a sub-alternative is found.
	## 
	## D y n a m i c   V a r i a b l e s  (- <name>: <class>; <description>):
	## none
	## 
	## R e s u l t s  (- <name>: <class>; <description>):
	## none
	## 
	## S i d e   E f f e c t s   (incl. important slot changes):
	## pcls are temporarily consumed. SEEK_RESULT is likely to be modified!
	## Free replicates may be moved out of the endangered zone.
	## 
	## C o n d i t i o n s  (- <name>: <type>; <indication>):
	## - ds_alternative_has_sub_alternatives: error; if one of the
	##     corresponding datasets has sub-alternatives, but
	##     ALLOW_SUB_TREE? is _false;
	## - swg_dsn_scheme_occupied: swg_dsn_error; if in one ore more
	##     of the alternatives another reader or writer is working.

	_local ds_view                   # database_view; an application dataset
	_local temp_repl                 # ds_version_view; temporary replicate
	_local scheme_ids << set.new()   # set of integer; keys of occupied schemes
	_local occupation_info           # list of {integer, string}; contains scheme-id/user-name pairs
	_local occupants << internal_text_output_stream.new()  # list of all occupants' login names
	_local occupied? << _false       # boolean; flag for occupation
	_local alt_path, repl_path       # rope of (_unset/string); alternative paths
	_local ds_info                   # swg_dsn_dataset_info; a dataset infomration record
	_local depth                     # integer; alternative path length
	_local a_scheme_path             # rope of integer; a replicate scheme path
	_local repl_list                 # list of {database_view,boolean}; a .replicates entry

	_for ds_name, found? _over seek_result.fast_keys_and_elements()
	_loop
		_if found?
		_then
			ds_view << gis_program_manager.cached_dataset(ds_name)
			# Find the deepest existing alternative corresponding to the
			# wanted path and split the path in an upper and a lower part,
			# where the upper part is expected to be an alternative path
			# and the lower part the remaining scheme path.
			# (The upper part should at least contain the design-top path.)
			ds_info << .dataset_infos.select(predicate.eq(:dataset_name,ds_name.as_charvec())).an_element()
			alt_path << _self.alt_path_from_scheme_path(ds_info,scheme_path)
			alt_path.add_first(_unset)
			
			temp_repl << ds_view.replicate_without_collections(0)
			temp_repl.dataset_name << ds_view.dataset_name
			
			_protect
				_if _self.go_to_path_if_possible(temp_repl,alt_path,:readonly)
				_then
					# The alternative to be deleted exists (and is the
					# current one in templ_repl)---so do the checking
					# (otherwise just skip, the caller will be informed
					# by leaving the _true entry in SEEK_RESULT unchanged).
					seek_result[ds_name] << alt_path
					_if _not allow_sub_tree? _andif
					    _not temp_repl.alternatives.empty?
					_then
						# We raise an error but with an abused argument:
						# Instead of the alternative name the dataset name is assigned.
						condition.raise(:ds_alternative_has_sub_alternatives,
								:name, ds_view.external_name)
# <==========================================================================================
					_endif
			
					occupation_info << _self.investigate_alternatives(temp_repl)
					_if _not occupation_info.empty?
					_then
						occupied? << _true
						occupants.write(newline_char, ds_view.external_name,": ")
						sep_str << ""
						_for occ _over occupation_info.fast_elements()
						_loop
							scheme_ids.add(occ[1])
							occupants.write(sep_str, occ[2])
							sep_str << ", "
						_endloop
					_endif
					# The alternative to be deleted does not exist---just skip
					# (the caller will be informed by leaving the _true entry
					# in SEEK_RESULT unchanged).
				_endif
			
			_protection
				temp_repl.discard()
			_endprotect

			_if occupied? _orif
			    (repl_list << .replicates[ds_name]) _is _unset
			_then
				_continue
	# ^------------------------------
			_endif
			
			# Now look for cached replicates, because these would not be
			# detected by the above mechanism ("yourself" may be reader in
			# as many replicates of a version as you like---it is reported
			# only once, and this is trivial, as you must be reader of the
			# version in order to get the current writer or different
			# readers information)
			depth << alt_path.size - 1 # ignore the _unset element
			_for repl_entry _over repl_list.fast_elements()
			_loop
				# if alt_path is a prefix of the replicate's path
				# then this is a conflict.
				repl_path << repl_entry[1].alternative_path
				_if repl_path.size >= depth _andif
				    alt_path.subseq(2,depth).eq?(repl_path.subseq(1,depth))
				_then
					# There is a conflict---but if the replicate
					# had been released we may escape.
					_if repl_entry[2] _is _true _andif  # free
					    depth >= 2
					_then
						# move the replicate out of the dangerous
						# zone. (note that alt_path.subseq(1,depth)
						# is one element shorter than alt_path)
						_self.do_move_replicate(
							repl_entry[1],
							alt_path.subseq(1,depth),
							:current)
					_else
						# The replicate seems still needed, so
						# give an alarm.
						occupied? << _true
						occupants.write(newline_char,
								ds_view.external_name,": ",
								system.user_name)
						# The scheme id may only be a guess since the
						# scheme alternative is likely to be one level
						# below the replicate's alternative path, but
						# better a rough guess than nothing.
						# To detect the very checkpoint of the
						# replicate could be an expensive way to
						# have a better guess, but this not true,
						# if the reference was yet :current. So
						# accept that it is not decidable.
						a_scheme_path <<
							_self.scheme_path_from_alt_path(ds_info,repl_path)
						_if a_scheme_path _isnt _unset _andif
						    _not a_scheme_path.empty?
						_then
							scheme_ids.add(a_scheme_path.last)
						_endif
					_endif
				_endif
			_endloop # _for repl_entry _over .replicates[ds_name]...
		_endif
	_endloop

	_if occupied?
	_then
		condition.raise(:swg_dsn_scheme_occupied,
				:scheme_keys, scheme_ids,
				:user, occupants.string)
# <======================================================
	_endif

_endmethod
$

#------------------------------------------------------------------------------
# Method: determine_reference_path()
#------------------------------------------------------------------------------
_pragma(classify_level=restricted, topic={swg, swg_design, dsn_user_alt_ctrl})
_private _method swg_dsn_user_alternative_control.determine_reference_path(reference, scheme_path)
	## D e s c r i p t i o n :
	## Returns that subsequence (a prefix section) of the given
	## SCHEME_PATH that corresponds to the given REFERENCE key
	## which is either a symbol or a relative level (see below).
	## This function is straight-forward:
	## With :this, the returned path will be same as SCHEME_PATH,
	## with :as_built_data, an empty path is returned, and with an
	## integer, the path is shortened by the given value.
	## 
	## A r g u m e n t s  (- <name>: <class>; <description>):
	## - REFERENCE: symbol/integer; if a symbol then it must be
	##     :this or :as_built_data, else it must be in the range
	##     between 0 and SCHEME_PATH.size.
	## - SCHEME_PATH: list of integer; a non-empty sequence of
	##     swg_dsn_scheme keys, according to the scheme path down
	##     from root (top scheme) to the very scheme; 
	##     a scheme path with first element 0, however, designates
	##     a build-plan pool alternative;
	## 
	## D y n a m i c   V a r i a b l e s  (- <name>: <class>; <description>):
	## none
	## 
	## R e s u l t s  (- <name>: <class>; <description>):
	## - REFERENCE_PATH: list of integer; a sub-list of SCHEME_PATH
	##     according to the given reference.
	## 
	## S i d e   E f f e c t s   (incl. important slot changes):
	## none
	## 
	## C o n d i t i o n s  (- <name>: <type>; <indication>):
	## none

	_local reference_path             # list of integer; the result (see above)
	
	_if reference.species _is 0.species
	_then
		reference_path << scheme_path.subseq(1, scheme_path.size - reference)
	_elif reference _is :this
	_then
		reference_path << scheme_path.copy()
	_elif reference _is :as_built_data
	_then
		_if scheme_path.first = 0
		_then
			# It is a build-plan pool; so set the build-plan top
			reference_path << {0}
		_else
			# It is a design scheme; so return the empty 
			reference_path << {}
		_endif
	_endif
	>> reference_path
_endmethod
$

#------------------------------------------------------------------------------
# Method: scheme_path_from_alt_path()
#------------------------------------------------------------------------------
_pragma(classify_level=restricted, topic={swg, swg_design, dsn_user_alt_ctrl})
_method swg_dsn_user_alternative_control.scheme_path_from_alt_path(dataset_info,alternative_path)
	## D e s c r i p t i o n :
	## Attempts to convert ALTERNATIVE_PATH, which is assumed to be
	## an alternative path of the dataset described by the dataset
	## information record DATASET_INFO (this is not checked!) to the
	## corresponding scheme path (being a sequence of cardinal numbers
	## representing id values of swg_dsn_scheme records) if possible.
	## In this case returns the scheme path, otherwise the result will
	## be _unset.
	## 
	## A r g u m e n t s  (- <name>: <class>; <description>):
	## - DATASET_INFO: swg_dsn_dataset_info; a valid dataset
	##     information record;
	## - ALTERNATIVE_PATH: list of string; the alternative path as
	##     yielded by ds_version_view.alternative_path. It is assumed
	##     to be absolute  w i t h o u t  a leading _unset element.
	## 
	## D y n a m i c   V a r i a b l e s  (- <name>: <class>; <description>):
	## 
	## !RELOCATING_SCHEME_ID!      either _UNSET or <a scheme_id>
	## !RELOCATING_TEMP_SCHEME_ID! either _UNSET or <a scheme_id>
	##
	## Both the above dynamics must be set, if ALTERNATIVE_PATH is
	## the path of an alternative for the still active temporary
	## scheme, with id !RELOCATING_TEMP_SCHEME_ID!, created in
	## order to relocate the scheme with id
	## !RELOCATING_SCHEME_ID! by a far-merge operation into the
	## new, non-pool managed, alternative specified by
	## ALTERNATIVE_PATH such that the alternative in its new
	## location will be named with an alternative name derived from
	## !RELOCATING_SCHEME_ID!, rather than the id of the still
	## current temporary scheme, !RELOCATING_TEMP_SCHEME_ID!.
	##
	## Until the the relocation has successfully completed, both of
	## these dynamics must be set whenever this method is called so
	## that the last element of the the returned SCHEME_PATH will
	## be the id of the temporary scheme rather than the scheme
	## that is still undergoing relocation. Refer to method
	## ALT_NAME_TO_ID() for further details).
	##
	## R e s u l t s  (- <name>: <class>; <description>):
	## - SCHEME_PATH: _unset / rope of integer; if the alternative
	##     path can be mapped to a potential scheme id sequence
	##     then this sequence, otherwise _unset. Note: SCHEME_PATH
	##     will be an empty rope if the alternative path corresponds
	##     to the design-top alternative of this dataset.
	##     If ALTERNATIVE_PATH is a path in the build plan pool
	##     branch then the first element of SCHEME_PATH will be 0.
	## 
	## S i d e   E f f e c t s   (incl. important slot changes):
	## none
	## 
	## C o n d i t i o n s  (- <name>: <type>; <indication>):
	## none

	_local scheme_path << rope.new()  # rope of integer; a list of swg_dsn_scheme keys
	_local scheme_id                  # integer; a potential swg_dsn_scheme id value
	_local reference_path << # rope of string; the path of the as-built data alternative,
	                         # but with preceding _unset element.
		_self.decompose_alternative_name(dataset_info.as_built_alternative_path)
	_local an_alt_name << dataset_info.design_top_alternative_name.as_charvec()


	reference_path.remove_nth(1)

	_if alternative_path.size <= reference_path.size
	_then
		# At this level it cannot be a scheme-related path
		_return _unset
# <===========================
	_endif

	# Decide by means of the sub-as-built-data level whether the
	# path may be in the design branch or in the build-plan pool
	# branch.
	_if alternative_path[reference_path.size + 1] = an_alt_name
	_then
		# It's the design branch---just add the design-top name to the
		# reference path.
		reference_path.add_last(an_alt_name)
	_elif alternative_path[reference_path.size + 1] =
	      (an_alt_name << dataset_info.old_build_plan_pool_name)
	_then
		# It's in the build-plan pool branch---so add the build plan
		# alternative name and mark the scheme path with an initial
		# zero entry.
		reference_path.add_last(an_alt_name.as_charvec())
		scheme_path.add_last(0)
	_else
		# It's neither this nor that, so it must be something vade but
		# irrelevant.
		_return _unset
# <===========================
	_endif

	# Now be exact and check also the intermediate alternative names.
	_for level _over 1.upto(reference_path.size)
	_loop
		_if alternative_path[level].lowercase ~= reference_path[level].lowercase
		_then
			# Obviously, we are in a different tree.
			_return _unset
# <===================================
		_endif
	_endloop

	# So far, we are in the design or build-plan branch and may
	# identify the related schemes now.

	ds_view << gis_program_manager.cached_dataset(dataset_info.dataset_name.as_symbol())

	_for level _over (reference_path.size + 1).upto(alternative_path.size)
	_loop
		_if (scheme_id << _self.alt_name_to_id(ds_view, alternative_path[level])) _is _unset
		_then
			# Syntactically, the alternative name does not correspond to a
			# scheme id. So reject the path.
			_return _unset
# <===================================
		_else
			scheme_path.add_last(scheme_id)
		_endif
	_endloop
	
	>> scheme_path
_endmethod
$

#------------------------------------------------------------------------------
# Method: alt_path_from_scheme_path()  KGU
#------------------------------------------------------------------------------
_pragma(classify_level=restricted, topic={swg, swg_design, dsn_user_alt_ctrl})
_method swg_dsn_user_alternative_control.alt_path_from_scheme_path(dataset_info,scheme_path)
	## D e s c r i p t i o n :
	## Attempts to convert SCHEME_PATH, which is assumed to be a sequence
	## of cardinal numbers representing id values of swg_dsn_scheme records,
	## to the corresponding alternative path of the dataset described by
	## the dataset information record DATASET_INFO (the alternative
	## existence is not checked!).
	## 
	## A r g u m e n t s  (- <name>: <class>; <description>):
	## - DATASET_INFO: swg_dsn_dataset_info; a valid dataset
	##     information record;
	## - SCHEME_PATH: rope of integer; Note: If SCHEME_PATH is
	##     an empty rope then the alternative path will correspond
	##     to the design-top alternative of this dataset.
	##     A scheme path with first element 0, however, designates
	##     a build-plan pool alternative in the old Pre-DM401
	##     location.
	## 
	## D y n a m i c   V a r i a b l e s
	## 
	## !RELOCATING_SCHEME_ID!      either _UNSET or <a scheme_id>
	## !RELOCATING_TEMP_SCHEME_ID! either _UNSET or <a scheme_id>
	##
	## Both the above dynamics must be set, if SCHEME_PATH is the
	## path of a still active temporary scheme, with id
	## !RELOCATING_TEMP_SCHEME_ID!, created in order to relocate
	## the scheme with id !RELOCATING_SCHEME_ID! by a far-merge
	## operation into a new, non-pool managed, alternative
	## structure created for SCHEME_PATH such that the scheme
	## alternatives in the new location will be named with 
	## their alternative name derived from
	## !RELOCATING_SCHEME_ID!, rather than SCHEM_PATH.LAST
	## This allows the original scheme to be linked to the new
	## alternative path when the temporary scheme is deleted on
	## completion of the far-merge operation. During the
	## relocation, both of these dynamics must be set so that the
	## alternative path for the temporary scheme is derived using
	## the id of the scheme being relocated. Refer to method
	## ID_TO_ALT_NAME() for further details).
	##
	## R e s u l t s  (- <name>: <class>; <description>):
	## - ALTERNATIVE_PATH: list of string; the alternative path as
	##     yielded by ds_version_view.alternative_path, hence
	##     without preceding _unset, though it is absolute.
	## 
	## S i d e   E f f e c t s   (incl. important slot changes):
	## none
	## 
	## C o n d i t i o n s  (- <name>: <type>; <indication>):
	## none

	# This method is for efficiency, but leaves the way of being
	# flexible with respect to the alternative names! In
	# particular, with this method, it will be impossible to have
	# renamed alternatives ("swg_dsn<nr>" --> "swg-dsn<nr>a" etc.)

	_local alternative_path << # rope of string; the path of the as-built data alternative,
	                           # but with preceding _unset element.
		_self.decompose_alternative_name(dataset_info.as_built_alternative_path)

	_if _not scheme_path.empty? _andif
	    scheme_path.first = 0
	_then
	# Build Plan in old location (which cannot be a pool managed alternative)
		bp_alt_name << dataset_info.old_build_plan_pool_name
		alternative_path.add_last(bp_alt_name.as_charvec())
		_if scheme_path.size > 1
		_then
			bps_id << scheme_path[2]
			alternative_path.add_last(write_string(_self.alt_name_prefix, bps_id))
		_else
			scheme_path << {}
		_endif
	_else 
		alternative_path.add_last(dataset_info.design_top_alternative_name.as_charvec())

		_for scheme_id _over scheme_path.fast_elements()
		_loop
			alternative_path.add_last(_self.id_to_alt_name(scheme_id, dataset_info.dataset_name))
		_endloop
	_endif
	
	alternative_path.remove_nth(1)

	>> alternative_path
_endmethod
$

#------------------------------------------------------------------------------
# Method: set_effective_scheme_path()
#------------------------------------------------------------------------------
_pragma(classify_level=restricted,
	topic={swg, swg_design, dsn_user_alt_ctrl})
_private _method swg_dsn_user_alternative_control.set_effective_scheme_path()
	## D e s c r i p t i o n :
	## Sets slot .effective_scheme_path to the longest path of all
	## scheme paths stored in .view_status entries, if these are
	## compatible, i.e. all are initial subsequences of the longest
	## one, otherwise sets the slot to _unset.
	## Returns a property list of all found scheme paths and access
	## modes, where the key is the internal dataset name and a
	## Kleenean information whetehr the paths are coherent or not.
	## 
	## A r g u m e n t s  (- <name>: <class>; <description>):
	## none
	## 
	## D y n a m i c   V a r i a b l e s  (- <name>: <class>; <description>):
	## none
	## 
	## R e s u l t s  (- <name>: <class>; <description>):
	## - ACTIVE_PATHS: property_list[symbol] of {list of integer, symbol};
	##     associates a vector of to values to each dataset name
	##     given by a DATASET_INFOS record:
	##     1. either a scheme path if the current alternative of
	##        the dataset is a design alternative or _unset if the
	##        dataset is not initialised or found or if the
	##        alternative cannot be translated to a scheem path;
	##     2. a symbol designating the current access mode:
	##        :none if there is no access (e.g. dataset missing),
	##        :write, :readonly, or :hard_readonly otherwise
	## - COHERENT??: kleenean; is _true if for all inquired
	##     datasets there are design alternatives and these are
	##     compatible; it is _false, if at least one dataset has an
	##     incompatible path, and it is _maybe if some datasets are
	##     not available but in the remaining datasets there are
	##     compatible paths.
	##     Note that a set of scheme paths is regarded as compatible
	##     if all of them are equal or are a prefix of the longest
	##     one of them, provided that shorter paths do not have
	##     :write mode, e.g. a set {[17,45,95,123], [17,45,95],
	##     [17,45,95,123], [17,45]} is compatible is this sense
	##     if paths [17,45] and [17,45,95] are readonly.
	## 
	## S i d e   E f f e c t s   (incl. important slot changes):
	## sets slot .effective_scheme_path
	## 
	## C o n d i t i o n s  (- <name>: <type>; <indication>):
	## none

	_local active_paths << property_list.new() 
	_local max_size << 0                 # integer; length of longest path 
	_local longest_path                  # list of integer; a scheme path (the longest one found)
	_local coherent?? << _true           # kleenean; result (see above)
	_local access_mode                   # symbol, access mode (one of :write, :readonly, :none)
	_local ds_view                       # gis_ds_view; a dataset
	_local scheme_path                   # rope of integer; a sequence of scheme ids

	_for ds_name, vs_entry _over .view_status.fast_keys_and_elements()
	_loop
		# vs_entry: swg_dsn_uac_view_status
		# ds_name: symbol, the dataset name

		ds_view << gis_program_manager.cached_dataset(ds_name)
		access_mode << :none
		scheme_path << _unset
		
		_if ds_view _isnt _unset _andif
		    (scheme_path << vs_entry.scheme_path) _isnt _unset
		_then
			access_mode << ds_view.mode
			_if scheme_path.size > max_size
			_then
				longest_path << scheme_path
				max_size << longest_path.size
			_endif
			active_paths[ds_name] << {scheme_path, access_mode}
		_else
			# Since at least one dataset is missing the result cannot be
			# better than _maybe.
			coherent?? << _maybe
		_endif # availability[ds_name] _is _true
		
	_endloop # for ds_name, vs_entry _over .view_status...

	# Now, having all scheme paths together, look for compatibility
	_if longest_path _isnt _unset 
	_then
		_for path_info _over active_paths.fast_elements()
		_loop
			scheme_path << path_info[1]
			_if ((scheme_path.size < max_size _andif path_info[2] _is :write) _orif  
			     _not scheme_path.eq?( longest_path.subseq(1,scheme_path.size) ))
			_then
				coherent?? << _false
				_leave
		# <===================
			_endif
		_endloop
	_endif

	_if coherent?? _isnt _false 
	_then
		.effective_scheme_path << longest_path
	_else
		.effective_scheme_path << _unset
	_endif
	
	>> active_paths, coherent??
_endmethod
$

#------------------------------------------------------------------------------
# Method: find_replicate_entry()
#------------------------------------------------------------------------------
_pragma(classify_level=restricted,
	topic={swg, swg_design, dsn_user_alt_ctrl})
_private _method swg_dsn_user_alternative_control.find_replicate_entry(repl_view,
								       _optional dataset_name)
	## D e s c r i p t i o n :
	## Seeks the given REPL_VIEW in the respective replicate list
	## of slot .replicates. If DATASET_NAME is given, then only
	## this list is inspected, otherwise all lists are examined.
	## If found, then returns the found entry vector, containing
	## REPL_VIEW as first element and the kleenean 'free' flag as
	## second element.
	## 
	## A r g u m e n t s  (- <name>: <class>; <description>):
	## - REPL_VIEW; database_view, the replicate to be looked for;
	## - DATASET_NAME: symbol; the internal dataset name.
	## 
	## D y n a m i c   V a r i a b l e s  (- <name>: <class>; <description>):
	## none
	## 
	## R e s u l t s  (- <name>: <class>; <description>):
	## - REPLICATE_ENTRY: _unset/{REPL_VIEW,<free??>} where <free??> is
	##     a Kleenean value being _true if the replicate isn't occupied.
	## 
	## S i d e   E f f e c t s   (incl. important slot changes):
	## none
	## 
	## C o n d i t i o n s  (- <name>: <type>; <indication>):
	## none

	_local replicate_entry           # {database_view,boolean}; the enry for REPL_VIEW
	                                 # in slot .replicates. First element is REPL_VIEW,
					 # second is _true if replicate is free.
	
	
	_if dataset_name _is _unset _andif
	    (dataset_name << _self.identify_dataset_of_replicate(repl_view)) _is _unset 
	_then
		# make an ultimate effort to look whether it is one of our
		# cached replicates such that at last the dataset can be
		# identified.
		_for ds_name, repl_list _over .replicates.fast_keys_and_elements()
		_loop @find_repl_outer
			# ds_name: symbol; the internal name of a dataset
			# repl_list: list of {<replicate>, <free??>}

			_for repl_entry _over repl_list.fast_elements()
			_loop
				# repl_entry: {<replicate>, <free??>}

				_if repl_entry[1] _is repl_view # identified
				_then
					dataset_name << ds_name
					replicate_entry << repl_entry
					_leave @find_repl_outer
		# <============================================
				_endif
				_if repl_entry[1].name = repl_view.name
				_then
					dataset_name << ds_name
				_endif
			_endloop
		_endloop
	_elif .replicates.includes_key?(dataset_name)
	_then
		_for repl_entry _over .replicates[dataset_name].fast_elements()
		_loop
			_if repl_entry[1] _is repl_view  # identified
			_then
				replicate_entry << repl_entry
				_leave
		# <===================
			_endif
		_endloop
	_endif

	>> replicate_entry, dataset_name
_endmethod

_pragma(classify_level=restricted, topic={swg, swg_design, dsn_admin, dsn_user_alt_ctrl})
_private _method swg_dsn_user_alternative_control.int!go_to_scheme_environment(scheme_path, dataset_infos, wanted_mode, _gather keys_and_args)
	##
	## Does the actual work for GO_TO_SCHEME_ENVIRONMENT() and
	## CREATE_SCHEME_ENVIRONMENT().
	##
	## See GO_TO_SCHEME_ENVIRONMENT() for further details.

	_dynamic !relocating_scheme_id!         # Set to the value of the optional parameter :relocating_scheme_id
		                                # If set, this will be the ID of the scheme being relocated via the
		                                # temporary scheme that has been specified by SCHEME_PATH
		                                # with ID !relocating_temp_scheme_id! (see method comment above)
	_dynamic !relocating_temp_scheme_id!    # If !relocating_scheme_id! has been set (see above) then this
		                                # will be set to the ID of the temporary scheme specified by SCHEME_PATH,
						# i.e. the value of SCHEME_PATH.LAST

	_local ds_name                            # symbol; the (internal) name of a dataset
	_local ds_view                            # database_view; the current dataset
	_local view_status                        # swg_dsn_uac_view_status; a .view_status element
	_local access_mode <<                     # property_list[symbol] of symbol; result
		property_list.new(dataset_infos.size) 
	_local at_target_version?                 # boolean; notes whether the requested
	                                          # alternative is reached.
	_local top_scheme? <<                     # _true if scheme_path is that of top-level scheme
		_self.top_level_scheme?(scheme_path)
	_local non_dc_managed_dsinfos <<     # subset for those which are not Design Clerk managed
	        _self.non_dc_managed_dataset_infos(dataset_infos)
	
	job_type_name << _if (sch << _self.scheme(scheme_path)) _isnt _unset
			 _then >> sch.project.job_type
			 _else >> ""
			 _endif 

	# Check the existence and accessibility of the required datasets
	# - found?: property_list[symbol] of boolean; dataset seek results:
	#     _false means that the dataset is missing;
	# - missing_string: character_vector; names of missing datasets;
	# - unsaved?: boolean; flag for unsaved changes.
	# May cause an error condition swd_dsn_design_system_not_working
	_local (found?, missing_string, unsaved?) <<
		_self.analyse_accessibility(dataset_infos)

	# Now raise initial warnings if necessary

	_if missing_string.size > 0
	_then
		condition.raise(:swg_dsn_missing_dataset, :datasets, missing_string)
	_endif

	_if unsaved?
	_then
	# Give a warning. If this returns, then we can go on. When returning to an updated dataset, a commit will be done.
		condition.raise(:swg_dsn_scheme_modified)
	_endif

	# Now initialise the return value

	
	_for ds_info _over .dataset_infos.fast_elements()
	_loop
		access_mode[ds_name] << _if found?[ds_name<<ds_info.dataset_name.as_symbol()] _is _unset _andif
					    (ds<<ds_info.referred_dataset) _isnt _unset
					_then # The dataset was not requested---just return the current mode
						
						>>_if ds.writable? _then >> :write _else >> :readonly _endif
					_elif (ds<<ds_info.referred_dataset) _isnt _unset
					_then # The dataset requested and checks for current mode
						
						>>  _if ds.writable? _then >> :write _else >> :readonly _endif
					_else # the dataset is not available
						
						>>:none
					_endif
	_endloop

	
	#
	# Start the effective task
	#

	creation_args << property_list.new_with(_scatter keys_and_args)
	add_missing_versions? << creation_args.remove_key(:add_missing_versions?).default(_true)
        !relocating_scheme_id! << creation_args.remove_key(:relocating_scheme_id)

	_if !relocating_scheme_id! _isnt _unset
	_then
	# SCHEME_PATH specifies a temporary scheme being used to
	# relocate the scheme with ID !relocating_scheme_id! - set
	# the dynamic !relocating_temp_scheme_id! to the ID of the
	# tempory scheme, as given by scheme_path.last
		!relocating_temp_scheme_id! << scheme_path.last
	_endif 
	
	# First pass to determine the design datasets that have a missing
	# alternative to be created and/or assigned from the free alternative pool

	ds_infos_for_alts_from_alt_pool << set.new()
	datasets_at_target_version? << property_list.new()

	_for ds_info _over dataset_infos.fast_elements()
	_loop
		_if (ds_view << ds_info.referred_dataset) _isnt _unset 
		_then
			ds_name << ds_view.dataset_name

			_if ds_view.updated?
			_then
				# Once arrived here, we are permitted to commit.
				ds_view.commit()
			_endif
			
			# Find the deepest existing alternative corresponding to the
			# wanted path and split the path in an upper and a lower part,
			# where the upper part is expected to be an alternative path
			# and the lower part the remaining scheme path.
			# (The upper part should at least contain the design-top path.)
			
			(upper,lower) << _self.go_as_deep_as_possible(ds_info,scheme_path,ds_view)
			
			_if upper.size + lower.size > _self.max_alt_level
			_then
				# Max. allowed alternative level exceeded - force an abort.
				condition.raise(:swg_dsn_max_alternative_level,:maximum,_self.max_alt_level)
			_endif

			_if lower.empty?
			_then
				datasets_at_target_version?[ds_name] << _true
			_else
				datasets_at_target_version?[ds_name] << _false
				_if top_scheme? _is _true _andif ds_info.dc_managed?? _is _false
				_then
					ds_infos_for_alts_from_alt_pool.add(ds_info)
				_endif
			_endif 
		_endif 
	_endloop 

	# Create and/or assign the required free alternatives from the alternative pool

	_if add_missing_versions? _andif _not ds_infos_for_alts_from_alt_pool.empty?
	_then
		ds_names << _self.dataset_names_for(ds_infos_for_alts_from_alt_pool)
		swg_dsn_alt_pool_manager.obtain_alternatives_for_design(scheme_path[1], ds_names, _false)
	_endif

	# Second pass to create any remaining missing alternatives, as
	# appropriate, and setup the return values.
	
	_for ds_info _over dataset_infos.fast_elements()
	_loop
		_if (ds_view << ds_info.referred_dataset) _isnt _unset 
		_then
			ds_name << ds_view.dataset_name
			at_target_version? << datasets_at_target_version?[ds_name]

			_if _not at_target_version?
			_then 
			# Find the deepest existing alternative corresponding to the
			# wanted path and split the path in an upper and a lower part,
			# where the upper part is expected to be an alternative path
			# and the lower part the remaining scheme path.
			# (The upper part should at least contain the design-top path.)
			
				(upper,lower) << _self.go_as_deep_as_possible(ds_info,scheme_path,ds_view)
				at_target_version? << lower.empty?
			_endif 
			
			# Create any remaining path

			_if _not at_target_version?
			_then
				_if add_missing_versions?
				_then
					# Raise a warning
					# (If the path does not completely exist, then allow the caller
					# to abort. Otherwise we will attempt to complete the path.)
					condition.raise(:swg_dsn_missing_alternative_do_create,
							:dataset_name,ds_view.external_name, :job_type_name, job_type_name)
					# Descending create the remaining path
					
					at_target_version? << _self.create_remaining_path(ds_view, lower, _scatter creation_args)
					
					_if _not at_target_version?
					_then
						found?[ds_name] << _false
						condition.raise(:swg_dsn_missing_alternative_create_failed,
								:dataset_name,ds_view.external_name)
					_endif 
				_else
					# Raise a warning
					# (If the path does not completely exist, then allow the caller
					# to abort. Otherwise we will attempt to complete the path.)
					condition.raise(:swg_dsn_missing_alternative_skip_create,
							:dataset_name,ds_view.external_name, :job_type_name, job_type_name)
					
				_endif
			_else
				# Note, that the alternative had already existed
				at_target_version? << _true
			_endif
			
			# Now look for final work on this dataset: If creation succeeded
			# then set the view status info, otherwise return to the design-top
			# alternative and withdraw write access. Also, in the case of an empty
			# scheme path, write access is not conceded, replicates are dropped

			view_status << .view_status[ds_name]
			view_status.current_alternative << ds_view.alternative_path.copy()

			_if at_target_version? _andif _not (scheme_path.empty? _orif scheme_path.eq?({0}))
			_then
				# The alternative in the current dataset had already existed
				# or has successfully been created.
				# Hence, concede the wanted mode (if achievable)
				ds_view.switch(wanted_mode)
				access_mode[ds_name] << ds_view.mode

				_if view_status.base_replicate_view _is _unset
				_then
					# As there hasn't been a replicate yet, just request one
					view_status.base_replicate_view << _self.do_request_replicate(ds_view)
				_endif
				# Move the existing replicate view to the corresponding base version
				_self.do_move_replicate(view_status.base_replicate_view, view_status.current_alternative, :base)

			_else
				# The design alternative is missing or at design_top or a build-plan scheme
				# Hence, deny write access, as we stick in an ancestor scheme or even the design_top:
				ds_view.switch(:readonly)
				access_mode[ds_name] << _if at_target_version?
							_then >> :readonly    # Design Top or build-plan
							_else >> :none        # Missing alternative
							_endif 
				
				_if view_status.base_replicate_view _isnt _unset
				_then
					# Give up the existing replicate view
					_self.do_release_replicate(view_status.base_replicate_view)
					view_status.base_replicate_view << _unset
				_endif
			_endif 
		_endif 
	_endloop 

	>> found?, access_mode
_endmethod 
$

_pragma(classify_level=restricted, topic={swg, swg_design, dsn_admin, dsn_user_alt_ctrl})
_method swg_dsn_user_alternative_control.update_active_base_replicates()

	## Updates the base replicate views for the currently active
	## scheme environment.
	##
	## This method should be called immediately following a merge
	## or post operation of the current scheme to update the cached
	## base replica views for the currently active scheme to the
	## base versions.

	_for vs_entry _over .view_status.fast_elements()
	_loop
		_if (base_repl << vs_entry.base_replicate_view) _isnt _unset
		_then
			_self.do_move_replicate(base_repl, vs_entry.current_alternative, :base)
		_endif
	_endloop 
_endmethod
$

_pragma(classify_level=restricted, topic={swg, swg_design, dsn_admin, dsn_user_alt_ctrl})
_method swg_dsn_user_alternative_control.active_base_replicate(dataset_name)

	## Returns the base replicate views for the currently active
	## scheme environment or _unset if none currently cached.

	>> _if (vs_entry << .view_status[dataset_name]) _isnt _unset
	   _then >> vs_entry.base_replicate_view
	   _else >> _unset
	   _endif 
_endmethod
